12def string_span_tokenize(s, sep): 

13 r""" 

14 Return the offsets of the tokens in *s*, as a sequence of ``(start, end)`` 

15 tuples, by splitting the string at each occurrence of *sep*. 

16 

17 >>> from nltk.tokenize.util import string_span_tokenize 

18 >>> s = '''Good muffins cost $3.88\nin New York. Please buy me 

19 ... two of them.\n\nThanks.''' 

20 >>> list(string_span_tokenize(s, " ")) 

21 [(0, 4), (5, 12), (13, 17), (18, 26), (27, 30), (31, 36), (37, 37), 

22 (38, 44), (45, 48), (49, 55), (56, 58), (59, 73)] 

23 

24 :param s: the string to be tokenized 

25 :type s: str 

26 :param sep: the token separator 

27 :type sep: str 

28 :rtype: iter(tuple(int, int)) 

29 """ 

30 if len(sep) == 0: 

31 raise ValueError("Token delimiter must not be empty") 

32 left = 0 

33 while True: 

34 try: 

35 right = s.index(sep, left) 

36 if right != 0: 

37 yield left, right 

38 except ValueError: 

39 if left != len(s): 

40 yield left, len(s) 

41 break 

42 

43 left = right + len(sep)

46def regexp_span_tokenize(s, regexp): 

47 r""" 

48 Return the offsets of the tokens in *s*, as a sequence of ``(start, end)`` 

49 tuples, by splitting the string at each successive match of *regexp*. 

50 

51 >>> from nltk.tokenize.util import regexp_span_tokenize 

52 >>> s = '''Good muffins cost $3.88\nin New York. Please buy me 

53 ... two of them.\n\nThanks.''' 

54 >>> list(regexp_span_tokenize(s, r'\s')) 

55 [(0, 4), (5, 12), (13, 17), (18, 23), (24, 26), (27, 30), (31, 36), 

56 (38, 44), (45, 48), (49, 51), (52, 55), (56, 58), (59, 64), (66, 73)] 

57 

58 :param s: the string to be tokenized 

59 :type s: str 

60 :param regexp: regular expression that matches token separators (must not be empty) 

61 :type regexp: str 

62 :rtype: iter(tuple(int, int)) 

63 """ 

64 left = 0 

65 for m in finditer(regexp, s): 

66 right, next = m.span() 

67 if right != left: 

68 yield left, right 

69 left = next 

70 yield left, len(s)

73def spans_to_relative(spans): 

74 r""" 

75 Return a sequence of relative spans, given a sequence of spans. 

76 

77 >>> from nltk.tokenize import WhitespaceTokenizer 

78 >>> from nltk.tokenize.util import spans_to_relative 

79 >>> s = '''Good muffins cost $3.88\nin New York. Please buy me 

80 ... two of them.\n\nThanks.''' 

81 >>> list(spans_to_relative(WhitespaceTokenizer().span_tokenize(s))) 

82 [(0, 4), (1, 7), (1, 4), (1, 5), (1, 2), (1, 3), (1, 5), (2, 6), 

83 (1, 3), (1, 2), (1, 3), (1, 2), (1, 5), (2, 7)] 

84 

85 :param spans: a sequence of (start, end) offsets of the tokens 

86 :type spans: iter(tuple(int, int)) 

87 :rtype: iter(tuple(int, int)) 

88 """ 

89 prev = 0 

90 for left, right in spans: 

91 yield left - prev, right - left 

92 prev = right

256def align_tokens(tokens, sentence): 

257 """ 

258 This module attempt to find the offsets of the tokens in *s*, as a sequence 

259 of ``(start, end)`` tuples, given the tokens and also the source string. 

260 

261 >>> from nltk.tokenize import TreebankWordTokenizer 

262 >>> from nltk.tokenize.util import align_tokens 

263 >>> s = str("The plane, bound for St Petersburg, crashed in Egypt's " 

264 ... "Sinai desert just 23 minutes after take-off from Sharm el-Sheikh " 

265 ... "on Saturday.") 

266 >>> tokens = TreebankWordTokenizer().tokenize(s) 

267 >>> expected = [(0, 3), (4, 9), (9, 10), (11, 16), (17, 20), (21, 23), 

268 ... (24, 34), (34, 35), (36, 43), (44, 46), (47, 52), (52, 54), 

269 ... (55, 60), (61, 67), (68, 72), (73, 75), (76, 83), (84, 89), 

270 ... (90, 98), (99, 103), (104, 109), (110, 119), (120, 122), 

271 ... (123, 131), (131, 132)] 

272 >>> output = list(align_tokens(tokens, s)) 

273 >>> len(tokens) == len(expected) == len(output) # Check that length of tokens and tuples are the same. 

274 True 

275 >>> expected == list(align_tokens(tokens, s)) # Check that the output is as expected. 

276 True 

277 >>> tokens == [s[start:end] for start, end in output] # Check that the slices of the string corresponds to the tokens. 

278 True 

279 

280 :param tokens: The list of strings that are the result of tokenization 

281 :type tokens: list(str) 

282 :param sentence: The original string 

283 :type sentence: str 

284 :rtype: list(tuple(int,int)) 

285 """ 

286 point = 0 

287 offsets = [] 

288 for token in tokens: 

289 try: 

290 start = sentence.index(token, point) 

291 except ValueError as e: 

292 raise ValueError(f'substring "{token}" not found in "{sentence}"') from e 

293 point = start + len(token) 

294 offsets.append((start, point)) 

295 return offsets
