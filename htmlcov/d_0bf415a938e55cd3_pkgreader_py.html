17class PackageReader(object): 

18 """ 

19 Provides access to the contents of a zip-format OPC package via its 

20 :attr:`serialized_parts` and :attr:`pkg_srels` attributes. 

21 """ 

22 def __init__(self, content_types, pkg_srels, sparts): 

23 super(PackageReader, self).__init__() 

24 self._pkg_srels = pkg_srels 

25 self._sparts = sparts 

26 

27 @staticmethod 

28 def from_file(pkg_file): 

29 """ 

30 Return a |PackageReader| instance loaded with contents of *pkg_file*. 

31 """ 

32 phys_reader = PhysPkgReader(pkg_file) 

33 content_types = _ContentTypeMap.from_xml(phys_reader.content_types_xml) 

34 pkg_srels = PackageReader._srels_for(phys_reader, PACKAGE_URI) 

35 sparts = PackageReader._load_serialized_parts( 

36 phys_reader, pkg_srels, content_types 

37 ) 

38 phys_reader.close() 

39 return PackageReader(content_types, pkg_srels, sparts) 

40 

41 def iter_sparts(self): 

42 """ 

43 Generate a 4-tuple `(partname, content_type, reltype, blob)` for each 

44 of the serialized parts in the package. 

45 """ 

46 for s in self._sparts: 

47 yield (s.partname, s.content_type, s.reltype, s.blob) 

48 

49 def iter_srels(self): 

50 """ 

51 Generate a 2-tuple `(source_uri, srel)` for each of the relationships 

52 in the package. 

53 """ 

54 for srel in self._pkg_srels: 

55 yield (PACKAGE_URI, srel) 

56 for spart in self._sparts: 

57 for srel in spart.srels: 

58 yield (spart.partname, srel)

60 @staticmethod 

61 def _load_serialized_parts(phys_reader, pkg_srels, content_types): 

62 """ 

63 Return a list of |_SerializedPart| instances corresponding to the 

64 parts in *phys_reader* accessible by walking the relationship graph 

65 starting with *pkg_srels*. 

66 """ 

67 sparts = [] 

68 part_walker = PackageReader._walk_phys_parts(phys_reader, pkg_srels) 

69 for partname, blob, reltype, srels in part_walker: 

70 content_type = content_types[partname] 

71 spart = _SerializedPart( 

72 partname, content_type, reltype, blob, srels 

73 ) 

74 sparts.append(spart) 

75 return tuple(sparts) 

76 

77 @staticmethod 

78 def _srels_for(phys_reader, source_uri): 

79 """ 

80 Return |_SerializedRelationships| instance populated with 

81 relationships for source identified by *source_uri*. 

82 """ 

83 rels_xml = phys_reader.rels_xml_for(source_uri) 

84 return _SerializedRelationships.load_from_xml( 

85 source_uri.baseURI, rels_xml) 

86 

87 @staticmethod 

88 def _walk_phys_parts(phys_reader, srels, visited_partnames=None): 

89 """ 

90 Generate a 4-tuple `(partname, blob, reltype, srels)` for each of the 

91 parts in *phys_reader* by walking the relationship graph rooted at 

92 srels. 

93 """ 

94 if visited_partnames is None: 

95 visited_partnames = [] 

96 for srel in srels: 

97 if srel.is_external: 

98 continue 

99 partname = srel.target_partname 

100 if partname in visited_partnames: 

101 continue 

102 visited_partnames.append(partname) 

103 reltype = srel.reltype 

104 part_srels = PackageReader._srels_for(phys_reader, partname) 

105 blob = phys_reader.blob_for(partname) 

106 yield (partname, blob, reltype, part_srels) 

107 next_walker = PackageReader._walk_phys_parts( 

108 phys_reader, part_srels, visited_partnames 

109 ) 

110 for partname, blob, reltype, srels in next_walker: 

111 yield (partname, blob, reltype, srels)

114class _ContentTypeMap(object): 

115 """ 

116 Value type providing dictionary semantics for looking up content type by 

117 part name, e.g. ``content_type = cti['/ppt/presentation.xml']``. 

118 """ 

119 def __init__(self): 

120 super(_ContentTypeMap, self).__init__() 

121 self._overrides = CaseInsensitiveDict() 

122 self._defaults = CaseInsensitiveDict() 

123 

124 def __getitem__(self, partname): 

125 """ 

126 Return content type for part identified by *partname*. 

127 """ 

128 if not isinstance(partname, PackURI): 

129 tmpl = "_ContentTypeMap key must be <type 'PackURI'>, got %s" 

130 raise KeyError(tmpl % type(partname)) 

131 if partname in self._overrides: 

132 return self._overrides[partname] 

133 if partname.ext in self._defaults: 

134 return self._defaults[partname.ext] 

135 tmpl = "no content type for partname '%s' in [Content_Types].xml" 

136 raise KeyError(tmpl % partname) 

137 

138 @staticmethod 

139 def from_xml(content_types_xml): 

140 """ 

141 Return a new |_ContentTypeMap| instance populated with the contents 

142 of *content_types_xml*. 

143 """ 

144 types_elm = parse_xml(content_types_xml) 

145 ct_map = _ContentTypeMap() 

146 for o in types_elm.overrides: 

147 ct_map._add_override(o.partname, o.content_type) 

148 for d in types_elm.defaults: 

149 ct_map._add_default(d.extension, d.content_type) 

150 return ct_map 

151 

152 def _add_default(self, extension, content_type): 

153 """ 

154 Add the default mapping of *extension* to *content_type* to this 

155 content type mapping. 

156 """ 

157 self._defaults[extension] = content_type 

158 

159 def _add_override(self, partname, content_type): 

160 """ 

161 Add the default mapping of *partname* to *content_type* to this 

162 content type mapping. 

163 """ 

164 self._overrides[partname] = content_type

286 @staticmethod 

287 def load_from_xml(baseURI, rels_item_xml): 

288 """ 

289 Return |_SerializedRelationships| instance loaded with the 

290 relationships contained in *rels_item_xml*. Returns an empty 

291 collection if *rels_item_xml* is |None|. 

292 """ 

293 srels = _SerializedRelationships() 

294 if rels_item_xml is not None: 

295 rels_elm = parse_xml(rels_item_xml)
