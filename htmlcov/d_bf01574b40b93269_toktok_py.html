10""" 

11The tok-tok tokenizer is a simple, general tokenizer, where the input has one 

12sentence per line; thus only final period is tokenized. 

13 

14Tok-tok has been tested on, and gives reasonably good results for English, 

15Persian, Russian, Czech, French, German, Vietnamese, Tajik, and a few others. 

16The input should be in UTF-8 encoding. 

17 

18Reference: 

19Jon Dehdari. 2014. A Neurophysiologically-Inspired Statistical Language 

20Model (Doctoral dissertation). Columbus, OH, USA: The Ohio State University. 

21""" 

22 

23import re 

24 

25from nltk.tokenize.api import TokenizerI 

26 

27 

28class ToktokTokenizer(TokenizerI): 

29 """ 

30 This is a Python port of the tok-tok.pl from 

31 https://github.com/jonsafari/tok-tok/blob/master/tok-tok.pl 

32 

33 >>> toktok = ToktokTokenizer() 

34 >>> text = u'Is 9.5 or 525,600 my favorite number?' 

35 >>> print(toktok.tokenize(text, return_str=True)) 

36 Is 9.5 or 525,600 my favorite number ? 

37 >>> text = u'The https://github.com/jonsafari/tok-tok/blob/master/tok-tok.pl is a website with/and/or slashes and sort of weird : things' 

38 >>> print(toktok.tokenize(text, return_str=True)) 

39 The https://github.com/jonsafari/tok-tok/blob/master/tok-tok.pl is a website with/and/or slashes and sort of weird : things 

40 >>> text = u'\xa1This, is a sentence with weird\xbb symbols\u2026 appearing everywhere\xbf' 

41 >>> expected = u'\xa1 This , is a sentence with weird \xbb symbols \u2026 appearing everywhere \xbf' 

42 >>> assert toktok.tokenize(text, return_str=True) == expected 

43 >>> toktok.tokenize(text) == [u'\xa1', u'This', u',', u'is', u'a', u'sentence', u'with', u'weird', u'\xbb', u'symbols', u'\u2026', u'appearing', u'everywhere', u'\xbf'] 

44 True 

45 """ 

46 

47 # Replace non-breaking spaces with normal spaces. 

48 NON_BREAKING = re.compile("\u00A0"), " " 

49 

50 # Pad some funky punctuation. 

51 FUNKY_PUNCT_1 = re.compile(r'([،;؛¿!"\])}»›”؟¡%٪°±©®।॥…])'), r" \1 " 

52 # Pad more funky punctuation. 

53 FUNKY_PUNCT_2 = re.compile(r"([({\[“‘„‚«‹「『])"), r" \1 " 

54 # Pad En dash and em dash 

55 EN_EM_DASHES = re.compile("([–—])"), r" \1 " 

56 

57 # Replace problematic character with numeric character reference. 

58 AMPERCENT = re.compile("& "), "&amp; " 

59 TAB = re.compile("\t"), " &#9; " 

60 PIPE = re.compile(r"\|"), " &#124; " 

61 

62 # Pad numbers with commas to keep them from further tokenization. 

63 COMMA_IN_NUM = re.compile(r"(?<!,)([,،])(?![,\d])"), r" \1 " 

64 

65 # Just pad problematic (often neurotic) hyphen/single quote, etc. 

66 PROB_SINGLE_QUOTES = re.compile(r"(['’`])"), r" \1 " 

67 # Group ` ` stupid quotes ' ' into a single token. 

68 STUPID_QUOTES_1 = re.compile(r" ` ` "), r" `` " 

69 STUPID_QUOTES_2 = re.compile(r" ' ' "), r" '' " 

70 

71 # Don't tokenize period unless it ends the line and that it isn't 

72 # preceded by another period, e.g. 

73 # "something ..." -> "something ..." 

74 # "something." -> "something ." 

75 FINAL_PERIOD_1 = re.compile(r"(?<!\.)\.$"), r" ." 

76 # Don't tokenize period unless it ends the line eg. 

77 # " ... stuff." -> "... stuff ." 

78 FINAL_PERIOD_2 = re.compile(r"""(?<!\.)\.\s*(["'’»›”]) *$"""), r" . \1" 

79 

80 # Treat continuous commas as fake German,Czech, etc.: „ 

81 MULTI_COMMAS = re.compile(r"(,{2,})"), r" \1 " 

82 # Treat continuous dashes as fake en-dash, etc. 

83 MULTI_DASHES = re.compile(r"(-{2,})"), r" \1 " 

84 # Treat multiple periods as a thing (eg. ellipsis) 

85 MULTI_DOTS = re.compile(r"(\.{2,})"), r" \1 " 

86 

87 # This is the \p{Open_Punctuation} from Perl's perluniprops 

88 # see https://perldoc.perl.org/perluniprops.html 

89 OPEN_PUNCT = str( 

90 "([{\u0f3a\u0f3c\u169b\u201a\u201e\u2045\u207d" 

91 "\u208d\u2329\u2768\u276a\u276c\u276e\u2770\u2772" 

92 "\u2774\u27c5\u27e6\u27e8\u27ea\u27ec\u27ee\u2983" 

93 "\u2985\u2987\u2989\u298b\u298d\u298f\u2991\u2993" 

94 "\u2995\u2997\u29d8\u29da\u29fc\u2e22\u2e24\u2e26" 

95 "\u2e28\u3008\u300a\u300c\u300e\u3010\u3014\u3016" 

96 "\u3018\u301a\u301d\ufd3e\ufe17\ufe35\ufe37\ufe39" 

97 "\ufe3b\ufe3d\ufe3f\ufe41\ufe43\ufe47\ufe59\ufe5b" 

98 "\ufe5d\uff08\uff3b\uff5b\uff5f\uff62" 

99 ) 

100 # This is the \p{Close_Punctuation} from Perl's perluniprops 

101 CLOSE_PUNCT = str( 

102 ")]}\u0f3b\u0f3d\u169c\u2046\u207e\u208e\u232a" 

103 "\u2769\u276b\u276d\u276f\u2771\u2773\u2775\u27c6" 

104 "\u27e7\u27e9\u27eb\u27ed\u27ef\u2984\u2986\u2988" 

105 "\u298a\u298c\u298e\u2990\u2992\u2994\u2996\u2998" 

106 "\u29d9\u29db\u29fd\u2e23\u2e25\u2e27\u2e29\u3009" 

107 "\u300b\u300d\u300f\u3011\u3015\u3017\u3019\u301b" 

108 "\u301e\u301f\ufd3f\ufe18\ufe36\ufe38\ufe3a\ufe3c" 

109 "\ufe3e\ufe40\ufe42\ufe44\ufe48\ufe5a\ufe5c\ufe5e" 

110 "\uff09\uff3d\uff5d\uff60\uff63" 

111 ) 

112 # This is the \p{Close_Punctuation} from Perl's perluniprops 

113 CURRENCY_SYM = str( 

114 "$\xa2\xa3\xa4\xa5\u058f\u060b\u09f2\u09f3\u09fb" 

115 "\u0af1\u0bf9\u0e3f\u17db\u20a0\u20a1\u20a2\u20a3" 

116 "\u20a4\u20a5\u20a6\u20a7\u20a8\u20a9\u20aa\u20ab" 

117 "\u20ac\u20ad\u20ae\u20af\u20b0\u20b1\u20b2\u20b3" 

118 "\u20b4\u20b5\u20b6\u20b7\u20b8\u20b9\u20ba\ua838" 

119 "\ufdfc\ufe69\uff04\uffe0\uffe1\uffe5\uffe6" 

120 ) 

121 

122 # Pad spaces after opening punctuations. 

123 OPEN_PUNCT_RE = re.compile(f"([{OPEN_PUNCT}])"), r"\1 " 

124 # Pad spaces before closing punctuations. 

125 CLOSE_PUNCT_RE = re.compile(f"([{CLOSE_PUNCT}])"), r"\1 " 

126 # Pad spaces after currency symbols. 

127 CURRENCY_SYM_RE = re.compile(f"([{CURRENCY_SYM}])"), r"\1 " 

128 

129 # Use for tokenizing URL-unfriendly characters: [:/?#] 

130 URL_FOE_1 = re.compile(r":(?!//)"), r" : " # in perl s{:(?!//)}{ : }g; 

131 URL_FOE_2 = re.compile(r"\?(?!\S)"), r" ? " # in perl s{\?(?!\S)}{ ? }g; 

132 # in perl: m{://} or m{\S+\.\S+/\S+} or s{/}{ / }g; 

133 URL_FOE_3 = re.compile(r"(:\/\/)[\S+\.\S+\/\S+][\/]"), " / " 

134 URL_FOE_4 = re.compile(r" /"), r" / " # s{ /}{ / }g; 

135 

136 # Left/Right strip, i.e. remove heading/trailing spaces. 

137 # These strip regexes should NOT be used, 

138 # instead use str.lstrip(), str.rstrip() or str.strip() 

139 # (They are kept for reference purposes to the original toktok.pl code) 

140 LSTRIP = re.compile(r"^ +"), "" 

141 RSTRIP = re.compile(r"\s+$"), "\n" 

142 # Merge multiple spaces. 

143 ONE_SPACE = re.compile(r" {2,}"), " " 

144 

145 TOKTOK_REGEXES = [ 

146 NON_BREAKING, 

147 FUNKY_PUNCT_1, 

148 URL_FOE_1, 

149 URL_FOE_2, 

150 URL_FOE_3, 

151 URL_FOE_4, 

152 AMPERCENT, 

153 TAB, 

154 PIPE, 

155 OPEN_PUNCT_RE, 

156 CLOSE_PUNCT_RE, 

157 MULTI_COMMAS, 

158 COMMA_IN_NUM, 

159 FINAL_PERIOD_2, 

160 PROB_SINGLE_QUOTES, 

161 STUPID_QUOTES_1, 

162 STUPID_QUOTES_2, 

163 CURRENCY_SYM_RE, 

164 EN_EM_DASHES, 

165 MULTI_DASHES, 

166 MULTI_DOTS, 

167 FINAL_PERIOD_1, 

168 FINAL_PERIOD_2, 

169 ONE_SPACE, 

170 ] 

171 

172 def tokenize(self, text, return_str=False): 

173 text = str(text) # Converts input string into unicode. 

174 for regexp, substitution in self.TOKTOK_REGEXES: 

175 text = regexp.sub(substitution, text) 

176 # Finally, strips heading and trailing spaces 

177 # and converts output string into unicode. 

178 text = str(text.strip()) 

179 return text if return_str else text.split()
