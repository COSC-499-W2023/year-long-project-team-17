15class AlignedSent: 

16 """ 

17 Return an aligned sentence object, which encapsulates two sentences 

18 along with an ``Alignment`` between them. 

19 

20 Typically used in machine translation to represent a sentence and 

21 its translation. 

22 

23 >>> from nltk.translate import AlignedSent, Alignment 

24 >>> algnsent = AlignedSent(['klein', 'ist', 'das', 'Haus'], 

25 ... ['the', 'house', 'is', 'small'], Alignment.fromstring('0-3 1-2 2-0 3-1')) 

26 >>> algnsent.words 

27 ['klein', 'ist', 'das', 'Haus'] 

28 >>> algnsent.mots 

29 ['the', 'house', 'is', 'small'] 

30 >>> algnsent.alignment 

31 Alignment([(0, 3), (1, 2), (2, 0), (3, 1)]) 

32 >>> from nltk.corpus import comtrans 

33 >>> print(comtrans.aligned_sents()[54]) 

34 <AlignedSent: 'Weshalb also sollten...' -> 'So why should EU arm...'> 

35 >>> print(comtrans.aligned_sents()[54].alignment) 

36 0-0 0-1 1-0 2-2 3-4 3-5 4-7 5-8 6-3 7-9 8-9 9-10 9-11 10-12 11-6 12-6 13-13 

37 

38 :param words: Words in the target language sentence 

39 :type words: list(str) 

40 :param mots: Words in the source language sentence 

41 :type mots: list(str) 

42 :param alignment: Word-level alignments between ``words`` and ``mots``. 

43 Each alignment is represented as a 2-tuple (words_index, mots_index). 

44 :type alignment: Alignment 

45 """ 

46 

47 def __init__(self, words, mots, alignment=None): 

48 self._words = words 

49 self._mots = mots 

50 if alignment is None: 

51 self.alignment = Alignment([]) 

52 else: 

53 assert type(alignment) is Alignment 

54 self.alignment = alignment 

55 

56 @property 

57 def words(self): 

58 return self._words 

59 

60 @property 

61 def mots(self): 

62 return self._mots 

63 

64 def _get_alignment(self): 

65 return self._alignment 

66 

67 def _set_alignment(self, alignment): 

68 _check_alignment(len(self.words), len(self.mots), alignment) 

69 self._alignment = alignment 

70 

71 alignment = property(_get_alignment, _set_alignment)

73 def __repr__(self): 

74 """ 

75 Return a string representation for this ``AlignedSent``. 

76 

77 :rtype: str 

78 """ 

79 words = "[%s]" % (", ".join("'%s'" % w for w in self._words)) 

80 mots = "[%s]" % (", ".join("'%s'" % w for w in self._mots)) 

81 

82 return f"AlignedSent({words}, {mots}, {self._alignment!r})"

84 def _to_dot(self): 

85 """ 

86 Dot representation of the aligned sentence 

87 """ 

88 s = "graph align {\n" 

89 s += "node[shape=plaintext]\n" 

90 

91 # Declare node 

92 for w in self._words: 

93 s += f'"{w}_source" [label="{w}"] \n' 

94 

95 for w in self._mots: 

96 s += f'"{w}_target" [label="{w}"] \n' 

97 

98 # Alignment 

99 for u, v in self._alignment: 

100 s += f'"{self._words[u]}_source" -- "{self._mots[v]}_target" \n' 

101 

102 # Connect the source words 

103 for i in range(len(self._words) - 1): 

104 s += '"{}_source" -- "{}_source" [style=invis]\n'.format( 

105 self._words[i], 

106 self._words[i + 1], 

107 ) 

108 

109 # Connect the target words 

110 for i in range(len(self._mots) - 1): 

111 s += '"{}_target" -- "{}_target" [style=invis]\n'.format( 

112 self._mots[i], 

113 self._mots[i + 1], 

114 ) 

115 

116 # Put it in the same rank 

117 s += "{rank = same; %s}\n" % (" ".join('"%s_source"' % w for w in self._words)) 

118 s += "{rank = same; %s}\n" % (" ".join('"%s_target"' % w for w in self._mots)) 

119 

120 s += "}" 

121 

122 return s

124 def _repr_svg_(self): 

125 """ 

126 Ipython magic : show SVG representation of this ``AlignedSent``. 

127 """ 

128 dot_string = self._to_dot().encode("utf8") 

129 output_format = "svg" 

130 try: 

131 process = subprocess.Popen( 

132 ["dot", "-T%s" % output_format], 

133 stdin=subprocess.PIPE, 

134 stdout=subprocess.PIPE, 

135 stderr=subprocess.PIPE, 

136 ) 

137 except OSError as e: 

138 raise Exception("Cannot find the dot binary from Graphviz package") from e 

139 out, err = process.communicate(dot_string) 

140 

141 return out.decode("utf8")

162class Alignment(frozenset): 

163 """ 

164 A storage class for representing alignment between two sequences, s1, s2. 

165 In general, an alignment is a set of tuples of the form (i, j, ...) 

166 representing an alignment between the i-th element of s1 and the 

167 j-th element of s2. Tuples are extensible (they might contain 

168 additional data, such as a boolean to indicate sure vs possible alignments). 

169 

170 >>> from nltk.translate import Alignment 

171 >>> a = Alignment([(0, 0), (0, 1), (1, 2), (2, 2)]) 

172 >>> a.invert() 

173 Alignment([(0, 0), (1, 0), (2, 1), (2, 2)]) 

174 >>> print(a.invert()) 

175 0-0 1-0 2-1 2-2 

176 >>> a[0] 

177 [(0, 1), (0, 0)] 

178 >>> a.invert()[2] 

179 [(2, 1), (2, 2)] 

180 >>> b = Alignment([(0, 0), (0, 1)]) 

181 >>> b.issubset(a) 

182 True 

183 >>> c = Alignment.fromstring('0-0 0-1') 

184 >>> b == c 

185 True 

186 """ 

187 

188 def __new__(cls, pairs): 

189 self = frozenset.__new__(cls, pairs) 

190 self._len = max(p[0] for p in self) if self != frozenset([]) else 0 

191 self._index = None 

192 return self 

193 

194 @classmethod 

195 def fromstring(cls, s): 

196 """ 

197 Read a giza-formatted string and return an Alignment object. 

198 

199 >>> Alignment.fromstring('0-0 2-1 9-2 21-3 10-4 7-5') 

200 Alignment([(0, 0), (2, 1), (7, 5), (9, 2), (10, 4), (21, 3)]) 

201 

202 :type s: str 

203 :param s: the positional alignments in giza format 

204 :rtype: Alignment 

205 :return: An Alignment object corresponding to the string representation ``s``. 

206 """ 

207 

208 return Alignment([_giza2pair(a) for a in s.split()]) 

209 

210 def __getitem__(self, key): 

211 """ 

212 Look up the alignments that map from a given index or slice. 

213 """ 

214 if not self._index: 

215 self._build_index() 

216 return self._index.__getitem__(key) 

217 

218 def invert(self): 

219 """ 

220 Return an Alignment object, being the inverted mapping. 

221 """ 

222 return Alignment(((p[1], p[0]) + p[2:]) for p in self) 

223 

224 def range(self, positions=None): 

225 """ 

226 Work out the range of the mapping from the given positions. 

227 If no positions are specified, compute the range of the entire mapping. 

228 """ 

229 image = set() 

230 if not self._index: 

231 self._build_index() 

232 if not positions: 

233 positions = list(range(len(self._index))) 

234 for p in positions: 

235 image.update(f for _, f in self._index[p]) 

236 return sorted(image)

238 def __repr__(self): 

239 """ 

240 Produce a Giza-formatted string representing the alignment. 

241 """ 

242 return "Alignment(%r)" % sorted(self) 

243 

244 def __str__(self): 

245 """ 

246 Produce a Giza-formatted string representing the alignment. 

247 """ 

248 return " ".join("%d-%d" % p[:2] for p in sorted(self))

250 def _build_index(self): 

251 """ 

252 Build a list self._index such that self._index[i] is a list 

253 of the alignments originating from word i. 

254 """ 

255 self._index = [[] for _ in range(self._len + 1)] 

256 for p in self: 

257 self._index[p[0]].append(p)

260def _giza2pair(pair_string): 

261 i, j = pair_string.split("-") 

262 return int(i), int(j)

264 

265def _naacl2pair(pair_string): 

266 i, j, p = pair_string.split("-") 

267 return int(i), int(j)

270def _check_alignment(num_words, num_mots, alignment): 

271 """ 

272 Check whether the alignments are legal. 

273 

274 :param num_words: the number of source language words 

275 :type num_words: int 

276 :param num_mots: the number of target language words 

277 :type num_mots: int 

278 :param alignment: alignment to be checked 

279 :type alignment: Alignment 

280 :raise IndexError: if alignment falls outside the sentence 

281 """ 

282 

283 assert type(alignment) is Alignment 

284 

285 if not all(0 <= pair[0] < num_words for pair in alignment): 

286 raise IndexError("Alignment is outside boundary of words") 

287 if not all(pair[1] is None or 0 <= pair[1] < num_mots for pair in alignment): 

288 raise IndexError("Alignment is outside boundary of mots")

294class PhraseTable: 

295 """ 

296 In-memory store of translations for a given phrase, and the log 

297 probability of the those translations 

298 """ 

299 

300 def __init__(self): 

301 self.src_phrases = dict() 

302 

303 def translations_for(self, src_phrase): 

304 """ 

305 Get the translations for a source language phrase 

306 

307 :param src_phrase: Source language phrase of interest 

308 :type src_phrase: tuple(str) 

309 

310 :return: A list of target language phrases that are translations 

311 of ``src_phrase``, ordered in decreasing order of 

312 likelihood. Each list element is a tuple of the target 

313 phrase and its log probability. 

314 :rtype: list(PhraseTableEntry) 

315 """ 

316 return self.src_phrases[src_phrase] 

317 

318 def add(self, src_phrase, trg_phrase, log_prob): 

319 """ 

320 :type src_phrase: tuple(str) 

321 :type trg_phrase: tuple(str) 

322 

323 :param log_prob: Log probability that given ``src_phrase``, 

324 ``trg_phrase`` is its translation 

325 :type log_prob: float 

326 """ 

327 entry = PhraseTableEntry(trg_phrase=trg_phrase, log_prob=log_prob) 

328 if src_phrase not in self.src_phrases: 

329 self.src_phrases[src_phrase] = [] 

330 self.src_phrases[src_phrase].append(entry) 

331 self.src_phrases[src_phrase].sort(key=lambda e: e.log_prob, reverse=True) 

332 

333 def __contains__(self, src_phrase): 

334 return src_phrase in self.src_phrases
