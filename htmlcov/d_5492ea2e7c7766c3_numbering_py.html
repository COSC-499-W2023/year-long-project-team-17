92class CT_Numbering(BaseOxmlElement): 

93 """ 

94 ``<w:numbering>`` element, the root element of a numbering part, i.e. 

95 numbering.xml 

96 """ 

97 num = ZeroOrMore('w:num', successors=('w:numIdMacAtCleanup',)) 

98 

99 def add_num(self, abstractNum_id): 

100 """ 

101 Return a newly added CT_Num (<w:num>) element referencing the 

102 abstract numbering definition identified by *abstractNum_id*. 

103 """ 

104 next_num_id = self._next_numId 

105 num = CT_Num.new(next_num_id, abstractNum_id) 

106 return self._insert_num(num) 

107 

108 def num_having_numId(self, numId): 

109 """ 

110 Return the ``<w:num>`` child element having ``numId`` attribute 

111 matching *numId*. 

112 """ 

113 xpath = './w:num[@w:numId="%d"]' % numId 

114 try: 

115 return self.xpath(xpath)[0] 

116 except IndexError: 

117 raise KeyError('no <w:num> element with numId %d' % numId) 

118 

119 @property 

120 def _next_numId(self): 

121 """ 

122 The first ``numId`` unused by a ``<w:num>`` element, starting at 

123 1 and filling any gaps in numbering between existing ``<w:num>`` 

124 elements. 

125 """ 

126 numId_strs = self.xpath('./w:num/@w:numId') 

127 num_ids = [int(numId_str) for numId_str in numId_strs] 

128 for num in range(1, len(num_ids)+2): 

129 if num not in num_ids: 

130 break 

131 return num
