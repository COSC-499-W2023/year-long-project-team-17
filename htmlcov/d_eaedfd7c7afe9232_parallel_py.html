142def _get_active_backend( 

143 prefer=default_parallel_config["prefer"], 

144 require=default_parallel_config["require"], 

145 verbose=default_parallel_config["verbose"], 

146): 

147 """Return the active default backend""" 

148 

149 backend_config = getattr(_backend, "config", default_parallel_config) 

150 

151 backend = _get_config_param( 

152 default_parallel_config['backend'], backend_config, "backend" 

153 ) 

154 prefer = _get_config_param(prefer, backend_config, "prefer") 

155 require = _get_config_param(require, backend_config, "require") 

156 verbose = _get_config_param(verbose, backend_config, "verbose") 

157 

158 if prefer not in VALID_BACKEND_HINTS: 

159 raise ValueError( 

160 f"prefer={prefer} is not a valid backend hint, " 

161 f"expected one of {VALID_BACKEND_HINTS}" 

162 ) 

163 if require not in VALID_BACKEND_CONSTRAINTS: 

164 raise ValueError( 

165 f"require={require} is not a valid backend constraint, " 

166 f"expected one of {VALID_BACKEND_CONSTRAINTS}" 

167 ) 

168 if prefer == 'processes' and require == 'sharedmem': 

169 raise ValueError( 

170 "prefer == 'processes' and require == 'sharedmem'" 

171 " are inconsistent settings" 

172 ) 

173 

174 explicit_backend = True 

175 if backend is None: 

176 

177 # We are either outside of the scope of any parallel_(config/backend) 

178 # context manager or the context manager did not set a backend. 

179 # create the default backend instance now. 

180 backend = BACKENDS[DEFAULT_BACKEND](nesting_level=0) 

181 explicit_backend = False 

182 

183 # Try to use the backend set by the user with the context manager. 

184 

185 nesting_level = backend.nesting_level 

186 uses_threads = getattr(backend, 'uses_threads', False) 

187 supports_sharedmem = getattr(backend, 'supports_sharedmem', False) 

188 # Force to use thread-based backend if the provided backend does not 

189 # match the shared memory constraint or if the backend is not explicitely 

190 # given and threads are prefered. 

191 force_threads = (require == 'sharedmem' and not supports_sharedmem) 

192 force_threads |= ( 

193 not explicit_backend and prefer == 'threads' and not uses_threads 

194 ) 

195 if force_threads: 

196 # This backend does not match the shared memory constraint: 

197 # fallback to the default thead-based backend. 

198 sharedmem_backend = BACKENDS[DEFAULT_THREAD_BACKEND]( 

199 nesting_level=nesting_level 

200 ) 

201 # Warn the user if we forced the backend to thread-based, while the 

202 # user explicitely specified a non-thread-based backend. 

203 if verbose >= 10 and explicit_backend: 

204 print( 

205 f"Using {sharedmem_backend.__class__.__name__} as " 

206 f"joblib backend instead of {backend.__class__.__name__} " 

207 "as the latter does not provide shared memory semantics." 

208 ) 

209 # Force to n_jobs=1 by default 

210 thread_config = backend_config.copy() 

211 thread_config['n_jobs'] = 1 

212 return sharedmem_backend, thread_config 

213 

214 return backend, backend_config

217class parallel_config: 

218 """Set the default backend or configuration for :class:`~joblib.Parallel`. 

219 

220 This is an alternative to directly passing keyword arguments to the 

221 :class:`~joblib.Parallel` class constructor. It is particularly useful when 

222 calling into library code that uses joblib internally but does not expose 

223 the various parallel configuration arguments in its own API. 

224 

225 Parameters 

226 ---------- 

227 backend : str or ParallelBackendBase instance, default=None 

228 If ``backend`` is a string it must match a previously registered 

229 implementation using the :func:`~register_parallel_backend` function. 

230 

231 By default the following backends are available: 

232 

233 - 'loky': single-host, process-based parallelism (used by default), 

234 - 'threading': single-host, thread-based parallelism, 

235 - 'multiprocessing': legacy single-host, process-based parallelism. 

236 

237 'loky' is recommended to run functions that manipulate Python objects. 

238 'threading' is a low-overhead alternative that is most efficient for 

239 functions that release the Global Interpreter Lock: e.g. I/O-bound 

240 code or CPU-bound code in a few calls to native code that explicitly 

241 releases the GIL. Note that on some rare systems (such as pyodide), 

242 multiprocessing and loky may not be available, in which case joblib 

243 defaults to threading. 

244 

245 In addition, if the ``dask`` and ``distributed`` Python packages are 

246 installed, it is possible to use the 'dask' backend for better 

247 scheduling of nested parallel calls without over-subscription and 

248 potentially distribute parallel calls over a networked cluster of 

249 several hosts. 

250 

251 It is also possible to use the distributed 'ray' backend for 

252 distributing the workload to a cluster of nodes. See more details 

253 in the Examples section below. 

254 

255 Alternatively the backend can be passed directly as an instance. 

256 

257 n_jobs : int, default=None 

258 The maximum number of concurrently running jobs, such as the number 

259 of Python worker processes when ``backend="loky"`` or the size of the 

260 thread-pool when ``backend="threading"``. 

261 If -1 all CPUs are used. If 1 is given, no parallel computing code 

262 is used at all, which is useful for debugging. For ``n_jobs`` below -1, 

263 (n_cpus + 1 + n_jobs) are used. Thus for ``n_jobs=-2``, all 

264 CPUs but one are used. 

265 ``None`` is a marker for 'unset' that will be interpreted as 

266 ``n_jobs=1`` in most backends. 

267 

268 verbose : int, default=0 

269 The verbosity level: if non zero, progress messages are 

270 printed. Above 50, the output is sent to stdout. 

271 The frequency of the messages increases with the verbosity level. 

272 If it more than 10, all iterations are reported. 

273 

274 temp_folder : str, default=None 

275 Folder to be used by the pool for memmapping large arrays 

276 for sharing memory with worker processes. If None, this will try in 

277 order: 

278 

279 - a folder pointed by the ``JOBLIB_TEMP_FOLDER`` environment 

280 variable, 

281 - ``/dev/shm`` if the folder exists and is writable: this is a 

282 RAM disk filesystem available by default on modern Linux 

283 distributions, 

284 - the default system temporary folder that can be 

285 overridden with ``TMP``, ``TMPDIR`` or ``TEMP`` environment 

286 variables, typically ``/tmp`` under Unix operating systems. 

287 

288 max_nbytes int, str, or None, optional, default='1M' 

289 Threshold on the size of arrays passed to the workers that 

290 triggers automated memory mapping in temp_folder. Can be an int 

291 in Bytes, or a human-readable string, e.g., '1M' for 1 megabyte. 

292 Use None to disable memmapping of large arrays. 

293 

294 mmap_mode: {None, 'r+', 'r', 'w+', 'c'}, default='r' 

295 Memmapping mode for numpy arrays passed to workers. None will 

296 disable memmapping, other modes defined in the numpy.memmap doc: 

297 https://numpy.org/doc/stable/reference/generated/numpy.memmap.html 

298 Also, see 'max_nbytes' parameter documentation for more details. 

299 

300 prefer: str in {'processes', 'threads'} or None, default=None 

301 Soft hint to choose the default backend. 

302 The default process-based backend is 'loky' and the default 

303 thread-based backend is 'threading'. Ignored if the ``backend`` 

304 parameter is specified. 

305 

306 require: 'sharedmem' or None, default=None 

307 Hard constraint to select the backend. If set to 'sharedmem', 

308 the selected backend will be single-host and thread-based. 

309 

310 inner_max_num_threads : int, default=None 

311 If not None, overwrites the limit set on the number of threads 

312 usable in some third-party library threadpools like OpenBLAS, 

313 MKL or OpenMP. This is only used with the ``loky`` backend. 

314 

315 backend_params : dict 

316 Additional parameters to pass to the backend constructor when 

317 backend is a string. 

318 

319 Notes 

320 ----- 

321 Joblib tries to limit the oversubscription by limiting the number of 

322 threads usable in some third-party library threadpools like OpenBLAS, MKL 

323 or OpenMP. The default limit in each worker is set to 

324 ``max(cpu_count() // effective_n_jobs, 1)`` but this limit can be 

325 overwritten with the ``inner_max_num_threads`` argument which will be used 

326 to set this limit in the child processes. 

327 

328 .. versionadded:: 1.3 

329 

330 Examples 

331 -------- 

332 >>> from operator import neg 

333 >>> with parallel_config(backend='threading'): 

334 ... print(Parallel()(delayed(neg)(i + 1) for i in range(5))) 

335 ... 

336 [-1, -2, -3, -4, -5] 

337 

338 To use the 'ray' joblib backend add the following lines: 

339 

340 >>> from ray.util.joblib import register_ray # doctest: +SKIP 

341 >>> register_ray() # doctest: +SKIP 

342 >>> with parallel_config(backend="ray"): # doctest: +SKIP 

343 ... print(Parallel()(delayed(neg)(i + 1) for i in range(5))) 

344 [-1, -2, -3, -4, -5]

666class BatchCompletionCallBack(object): 

667 """Callback to keep track of completed results and schedule the next tasks. 

668 

669 This callable is executed by the parent process whenever a worker process 

670 has completed a batch of tasks. 

671 

672 It is used for progress reporting, to update estimate of the batch 

673 processing duration and to schedule the next batch of tasks to be 

674 processed. 

675 

676 It is assumed that this callback will always be triggered by the backend 

677 right after the end of a task, in case of success as well as in case of 

678 failure. 

679 """ 

680 

681 ########################################################################## 

682 # METHODS CALLED BY THE MAIN THREAD # 

683 ########################################################################## 

684 def __init__(self, dispatch_timestamp, batch_size, parallel): 

685 self.dispatch_timestamp = dispatch_timestamp 

686 self.batch_size = batch_size 

687 self.parallel = parallel 

688 self.parallel_call_id = parallel._call_id 

689 

690 # Internals to keep track of the status and outcome of the task. 

691 

692 # Used to hold a reference to the future-like object returned by the 

693 # backend after launching this task 

694 # This will be set later when calling `register_job`, as it is only 

695 # created once the task has been submitted. 

696 self.job = None 

697 

698 if not parallel._backend.supports_retrieve_callback: 

699 # The status is only used for asynchronous result retrieval in the 

700 # callback. 

701 self.status = None 

702 else: 

703 # The initial status for the job is TASK_PENDING. 

704 # Once it is done, it will be either TASK_DONE, or TASK_ERROR. 

705 self.status = TASK_PENDING 

706 

707 def register_job(self, job): 

708 """Register the object returned by `apply_async`.""" 

709 self.job = job 

710 

711 def get_result(self, timeout): 

712 """Returns the raw result of the task that was submitted. 

713 

714 If the task raised an exception rather than returning, this same 

715 exception will be raised instead. 

716 

717 If the backend supports the retrieval callback, it is assumed that this 

718 method is only called after the result has been registered. It is 

719 ensured by checking that `self.status(timeout)` does not return 

720 TASK_PENDING. In this case, `get_result` directly returns the 

721 registered result (or raise the registered exception). 

722 

723 For other backends, there are no such assumptions, but `get_result` 

724 still needs to synchronously retrieve the result before it can 

725 return it or raise. It will block at most `self.timeout` seconds 

726 waiting for retrieval to complete, after that it raises a TimeoutError. 

727 """ 

728 

729 backend = self.parallel._backend 

730 

731 if backend.supports_retrieve_callback: 

732 # We assume that the result has already been retrieved by the 

733 # callback thread, and is stored internally. It's just waiting to 

734 # be returned. 

735 return self._return_or_raise() 

736 

737 # For other backends, the main thread needs to run the retrieval step. 

738 try: 

739 if backend.supports_timeout: 

740 result = self.job.get(timeout=timeout) 

741 else: 

742 result = self.job.get() 

743 outcome = dict(result=result, status=TASK_DONE) 

744 except BaseException as e: 

745 outcome = dict(result=e, status=TASK_ERROR) 

746 self._register_outcome(outcome) 

747 

748 return self._return_or_raise() 

749 

750 def _return_or_raise(self): 

751 try: 

752 if self.status == TASK_ERROR: 

753 raise self._result 

754 return self._result 

755 finally: 

756 del self._result 

757 

758 def get_status(self, timeout): 

759 """Get the status of the task. 

760 

761 This function also checks if the timeout has been reached and register 

762 the TimeoutError outcome when it is the case. 

763 """ 

764 if timeout is None or self.status != TASK_PENDING: 

765 return self.status 

766 

767 # The computation are running and the status is pending. 

768 # Check that we did not wait for this jobs more than `timeout`. 

769 now = time.time() 

770 if not hasattr(self, "_completion_timeout_counter"): 

771 self._completion_timeout_counter = now 

772 

773 if (now - self._completion_timeout_counter) > timeout: 

774 outcome = dict(result=TimeoutError(), status=TASK_ERROR) 

775 self._register_outcome(outcome) 

776 

777 return self.status 

778 

779 ########################################################################## 

780 # METHODS CALLED BY CALLBACK THREADS # 

781 ########################################################################## 

782 def __call__(self, out): 

783 """Function called by the callback thread after a job is completed.""" 

784 

785 # If the backend doesn't support callback retrievals, the next batch of 

786 # tasks is dispatched regardless. The result will be retrieved by the 

787 # main thread when calling `get_result`. 

788 if not self.parallel._backend.supports_retrieve_callback: 

789 self._dispatch_new() 

790 return 

791 

792 # If the backend supports retrieving the result in the callback, it 

793 # registers the task outcome (TASK_ERROR or TASK_DONE), and schedules 

794 # the next batch if needed. 

795 with self.parallel._lock: 

796 # Edge case where while the task was processing, the `parallel` 

797 # instance has been reset and a new call has been issued, but the 

798 # worker managed to complete the task and trigger this callback 

799 # call just before being aborted by the reset. 

800 if self.parallel._call_id != self.parallel_call_id: 

801 return 

802 

803 # When aborting, stop as fast as possible and do not retrieve the 

804 # result as it won't be returned by the Parallel call. 

805 if self.parallel._aborting: 

806 return 

807 

808 # Retrieves the result of the task in the main process and dispatch 

809 # a new batch if needed. 

810 job_succeeded = self._retrieve_result(out) 

811 

812 if job_succeeded: 

813 self._dispatch_new() 

814 

815 def _dispatch_new(self): 

816 """Schedule the next batch of tasks to be processed.""" 

817 

818 # This steps ensure that auto-baching works as expected. 

819 this_batch_duration = time.time() - self.dispatch_timestamp 

820 self.parallel._backend.batch_completed(self.batch_size, 

821 this_batch_duration) 

822 

823 # Schedule the next batch of tasks. 

824 with self.parallel._lock: 

825 self.parallel.n_completed_tasks += self.batch_size 

826 self.parallel.print_progress() 

827 if self.parallel._original_iterator is not None: 

828 self.parallel.dispatch_next() 

829 

830 def _retrieve_result(self, out): 

831 """Fetch and register the outcome of a task. 

832 

833 Return True if the task succeeded, False otherwise. 

834 This function is only called by backends that support retrieving 

835 the task result in the callback thread. 

836 """ 

837 try: 

838 result = self.parallel._backend.retrieve_result_callback(out) 

839 outcome = dict(status=TASK_DONE, result=result) 

840 except BaseException as e: 

841 # Avoid keeping references to parallel in the error. 

842 e.__traceback__ = None 

843 outcome = dict(result=e, status=TASK_ERROR) 

844 

845 self._register_outcome(outcome) 

846 return outcome['status'] != TASK_ERROR 

847 

848 ########################################################################## 

849 # This method can be called either in the main thread # 

850 # or in the callback thread. # 

851 ########################################################################## 

852 def _register_outcome(self, outcome): 

853 """Register the outcome of a task. 

854 

855 This method can be called only once, future calls will be ignored. 

856 """ 

857 # Covers the edge case where the main thread tries to register a 

858 # `TimeoutError` while the callback thread tries to register a result 

859 # at the same time. 

860 with self.parallel._lock: 

861 if self.status not in (TASK_PENDING, None): 

862 return 

863 self.status = outcome["status"] 

864 

865 self._result = outcome["result"] 

866 

867 # Once the result and the status are extracted, the last reference to 

868 # the job can be deleted. 

869 self.job = None 

870 

871 # As soon as an error as been spotted, early stopping flags are sent to 

872 # the `parallel` instance. 

873 if self.status == TASK_ERROR: 

874 self.parallel._exception = True 

875 self.parallel._aborting = True

1391 def dispatch_one_batch(self, iterator): 

1392 """Prefetch the tasks for the next batch and dispatch them. 

1393 

1394 The effective size of the batch is computed here. 

1395 If there are no more jobs to dispatch, return False, else return True. 

1396 

1397 The iterator consumption and dispatching is protected by the same 

1398 lock so calling this function should be thread safe. 

1399 

1400 """ 

1401 

1402 if self._aborting: 

1403 return False 

1404 

1405 batch_size = self._get_batch_size() 

1406 

1407 with self._lock: 

1408 # to ensure an even distribution of the workload between workers, 

1409 # we look ahead in the original iterators more than batch_size 

1410 # tasks - However, we keep consuming only one batch at each 

1411 # dispatch_one_batch call. The extra tasks are stored in a local 

1412 # queue, _ready_batches, that is looked-up prior to re-consuming 

1413 # tasks from the origal iterator. 

1414 try: 

1415 tasks = self._ready_batches.get(block=False) 

1416 except queue.Empty: 

1417 # slice the iterator n_jobs * batchsize items at a time. If the 

1418 # slice returns less than that, then the current batchsize puts 

1419 # too much weight on a subset of workers, while other may end 

1420 # up starving. So in this case, re-scale the batch size 

1421 # accordingly to distribute evenly the last items between all 

1422 # workers. 

1423 n_jobs = self._cached_effective_n_jobs 

1424 big_batch_size = batch_size * n_jobs 

1425 

1426 islice = list(itertools.islice(iterator, big_batch_size)) 

1427 if len(islice) == 0: 

1428 return False 

1429 elif (iterator is self._original_iterator and 

1430 len(islice) < big_batch_size): 

1431 # We reached the end of the original iterator (unless 

1432 # iterator is the ``pre_dispatch``-long initial slice of 

1433 # the original iterator) -- decrease the batch size to 

1434 # account for potential variance in the batches running 

1435 # time. 

1436 final_batch_size = max(1, len(islice) // (10 * n_jobs)) 

1437 else: 

1438 final_batch_size = max(1, len(islice) // n_jobs) 

1439 

1440 # enqueue n_jobs batches in a local queue 

1441 for i in range(0, len(islice), final_batch_size): 

1442 tasks = BatchedCalls(islice[i:i + final_batch_size], 

1443 self._backend.get_nested_backend(), 

1444 self._reducer_callback, 

1445 self._pickle_cache) 

1446 self._ready_batches.put(tasks) 

1447 

1448 # finally, get one task. 

1449 tasks = self._ready_batches.get(block=False) 

1450 if len(tasks) == 0: 

1451 # No more tasks available in the iterator: tell caller to stop. 

1452 return False 

1453 else: 

1454 self._dispatch(tasks) 

1455 return True

1483 def print_progress(self): 

1484 """Display the process of the parallel execution only a fraction 

1485 of time, controlled by self.verbose. 

1486 """ 

1487 

1488 if not self.verbose: 

1489 return 

1490 

1491 elapsed_time = time.time() - self._start_time 

1492 

1493 if self._is_completed(): 

1494 # Make sure that we get a last message telling us we are done 

1495 self._print( 

1496 f"Done {self.n_completed_tasks:3d} out of " 

1497 f"{self.n_completed_tasks:3d} | elapsed: " 

1498 f"{short_format_time(elapsed_time)} finished" 

1499 ) 

1500 return 

1501 

1502 # Original job iterator becomes None once it has been fully 

1503 # consumed : at this point we know the total number of jobs and we are 

1504 # able to display an estimation of the remaining time based on already 

1505 # completed jobs. Otherwise, we simply display the number of completed 

1506 # tasks. 

1507 elif self._original_iterator is not None: 

1508 if _verbosity_filter(self.n_dispatched_batches, self.verbose): 

1509 return

1511 f"Done {self.n_completed_tasks:3d} tasks | elapsed: " 

1512 f"{short_format_time(elapsed_time)}" 

1513 ) 

1514 else: 

1515 index = self.n_completed_tasks 

1516 # We are finished dispatching 

1517 total_tasks = self.n_dispatched_tasks 

1518 # We always display the first loop 

1519 if not index == 0: 

1520 # Display depending on the number of remaining items 

1521 # A message as soon as we finish dispatching, cursor is 0 

1522 cursor = (total_tasks - index + 1 - 

1523 self._pre_dispatch_amount) 

1524 frequency = (total_tasks // self.verbose) + 1 

1525 is_last_item = (index + 1 == total_tasks) 

1526 if (is_last_item or cursor % frequency): 

1527 return 

1528 remaining_time = (elapsed_time / index) * \ 

1529 (self.n_dispatched_tasks - index * 1.0) 

1530 # only display status if remaining time is greater or equal to 0 

1531 self._print( 

1532 f"Done {index:3d} out of {total_tasks:3d} | elapsed: " 

1533 f"{short_format_time(elapsed_time)} remaining: " 

1534 f"{short_format_time(remaining_time)}" 

1535 )

1554 def _start(self, iterator, pre_dispatch): 

1555 # Only set self._iterating to True if at least a batch 

1556 # was dispatched. In particular this covers the edge 

1557 # case of Parallel used with an exhausted iterator. If 

1558 # self._original_iterator is None, then this means either 

1559 # that pre_dispatch == "all", n_jobs == 1 or that the first batch 

1560 # was very quick and its callback already dispatched all the 

1561 # remaining jobs. 

1562 self._iterating = False 

1563 if self.dispatch_one_batch(iterator): 

1564 self._iterating = self._original_iterator is not None 

1565 

1566 while self.dispatch_one_batch(iterator): 

1567 pass 

1568 

1569 if pre_dispatch == "all": 

1570 # The iterable was consumed all at once by the above for loop. 

1571 # No need to wait for async callbacks to trigger to 

1572 # consumption. 

1573 self._iterating = False

1575 def _get_outputs(self, iterator, pre_dispatch): 

1576 """Iterator returning the tasks' output as soon as they are ready.""" 

1577 dispatch_thread_id = threading.get_ident() 

1578 detach_generator_exit = False 

1579 try: 

1580 self._start(iterator, pre_dispatch) 

1581 # first yield returns None, for internal use only. This ensures 

1582 # that we enter the try/except block and start dispatching the 

1583 # tasks. 

1584 yield 

1585 

1586 with self._backend.retrieval_context(): 

1587 yield from self._retrieve() 

1588 

1589 except GeneratorExit: 

1590 # The generator has been garbage collected before being fully 

1591 # consumed. This aborts the remaining tasks if possible and warn 

1592 # the user if necessary. 

1593 self._exception = True 

1594 

1595 # In some interpreters such as PyPy, GeneratorExit can be raised in 

1596 # a different thread than the one used to start the dispatch of the 

1597 # parallel tasks. This can lead to hang when a thread attempts to 

1598 # join itself. As workaround, we detach the execution of the 

1599 # aborting code to a dedicated thread. We then need to make sure 

1600 # the rest of the function does not call `_terminate_and_reset` 

1601 # in finally. 

1602 if dispatch_thread_id != threading.get_ident(): 

1603 if not IS_PYPY: 

1604 warnings.warn( 

1605 "A generator produced by joblib.Parallel has been " 

1606 "gc'ed in an unexpected thread. This behavior should " 

1607 "not cause major -issues but to make sure, please " 

1608 "report this warning and your use case at " 

1609 "https://github.com/joblib/joblib/issues so it can " 

1610 "be investigated." 

1611 ) 

1612 

1613 detach_generator_exit = True 

1614 _parallel = self 

1615 

1616 class _GeneratorExitThread(threading.Thread): 

1617 def run(self): 

1618 _parallel._abort() 

1619 if _parallel.return_generator: 

1620 _parallel._warn_exit_early() 

1621 _parallel._terminate_and_reset() 

1622 

1623 _GeneratorExitThread( 

1624 name="GeneratorExitThread" 

1625 ).start() 

1626 return 

1627 

1628 # Otherwise, we are in the thread that started the dispatch: we can 

1629 # safely abort the execution and warn the user. 

1630 self._abort() 

1631 if self.return_generator: 

1632 self._warn_exit_early() 

1633 

1634 raise 

1635 

1636 # Note: we catch any BaseException instead of just Exception instances 

1637 # to also include KeyboardInterrupt 

1638 except BaseException: 

1639 self._exception = True 

1640 self._abort() 

1641 raise 

1642 finally: 

1643 # Store the unconsumed tasks and terminate the workers if necessary 

1644 _remaining_outputs = ([] if self._exception else self._jobs) 

1645 self._jobs = collections.deque() 

1646 self._running = False 

1647 if not detach_generator_exit: 

1648 self._terminate_and_reset() 

1649 

1650 while len(_remaining_outputs) > 0: 

1651 batched_results = _remaining_outputs.popleft() 

1652 batched_results = batched_results.get_result(self.timeout) 

1653 for result in batched_results: 

1654 yield result

1656 def _wait_retrieval(self): 

1657 """Return True if we need to continue retriving some tasks.""" 

1658 

1659 # If the input load is still being iterated over, it means that tasks 

1660 # are still on the dispatch wait list and their results will need to 

1661 # be retrieved later on. 

1662 if self._iterating: 

1663 return True 

1664 

1665 # If some of the dispatched tasks are still being processed by the 

1666 # workers, wait for the compute to finish before starting retrieval 

1667 if self.n_completed_tasks < self.n_dispatched_tasks: 

1668 return True 

1669 

1670 # For backends that does not support retrieving asynchronously the 

1671 # result to the main process, all results must be carefully retrieved 

1672 # in the _retrieve loop in the main thread while the backend is alive. 

1673 # For other backends, the actual retrieval is done asynchronously in 

1674 # the callback thread, and we can terminate the backend before the 

1675 # `self._jobs` result list has been emptied. The remaining results 

1676 # will be collected in the `finally` step of the generator. 

1677 if not self._backend.supports_retrieve_callback: 

1678 if len(self._jobs) > 0: 

1679 return True 

1680 

1681 return False

1683 def _retrieve(self): 

1684 while self._wait_retrieval(): 

1685 

1686 # If the callback thread of a worker has signaled that its task 

1687 # triggered an exception, or if the retrieval loop has raised an 

1688 # exception (e.g. `GeneratorExit`), exit the loop and surface the 

1689 # worker traceback. 

1690 if self._aborting: 

1691 self._raise_error_fast() 

1692 break 

1693 

1694 # If the next job is not ready for retrieval yet, we just wait for 

1695 # async callbacks to progress. 

1696 if ((len(self._jobs) == 0) or 

1697 (self._jobs[0].get_status( 

1698 timeout=self.timeout) == TASK_PENDING)): 

1699 time.sleep(0.01) 

1700 continue 

1701 

1702 # We need to be careful: the job list can be filling up as 

1703 # we empty it and Python list are not thread-safe by 

1704 # default hence the use of the lock 

1705 with self._lock: 

1706 batched_results = self._jobs.popleft() 

1707 

1708 # Flatten the batched results to output one output at a time 

1709 batched_results = batched_results.get_result(self.timeout) 

1710 for result in batched_results: 

1711 self._nb_consumed += 1 

1712 yield result

1714 def _raise_error_fast(self): 

1715 """If we are aborting, raise if a job caused an error.""" 

1716 

1717 # Find the first job whose status is TASK_ERROR if it exists. 

1718 with self._lock: 

1719 error_job = next((job for job in self._jobs 

1720 if job.status == TASK_ERROR), None) 

1721 

1722 # If this error job exists, immediatly raise the error by 

1723 # calling get_result. This job might not exists if abort has been 

1724 # called directly or if the generator is gc'ed. 

1725 if error_job is not None: 

1726 error_job.get_result(self.timeout)

1728 def _warn_exit_early(self): 

1729 """Warn the user if the generator is gc'ed before being consumned.""" 

1730 ready_outputs = self.n_completed_tasks - self._nb_consumed 

1731 is_completed = self._is_completed() 

1732 msg = "" 

1733 if ready_outputs: 

1734 msg += ( 

1735 f"{ready_outputs} tasks have been successfully executed " 

1736 " but not used." 

1737 ) 

1738 if not is_completed: 

1739 msg += " Additionally, " 

1740 

1741 if not is_completed: 

1742 msg += ( 

1743 f"{self.n_dispatched_tasks - self.n_completed_tasks} tasks " 

1744 "which were still being processed by the workers have been " 

1745 "cancelled." 

1746 ) 

1747 

1748 if msg: 

1749 msg += ( 

1750 " You could benefit from adjusting the input task " 

1751 "iterator to limit unnecessary computation time." 

1752 ) 

1753 

1754 warnings.warn(msg)

1756 def _get_sequential_output(self, iterable): 

1757 """Separate loop for sequential output. 

1758 

1759 This simplifies the traceback in case of errors and reduces the 

1760 overhead of calling sequential tasks with `joblib`. 

1761 """ 

1762 try: 

1763 self._iterating = True 

1764 self._original_iterator = iterable 

1765 batch_size = self._get_batch_size() 

1766 

1767 if batch_size != 1: 

1768 it = iter(iterable) 

1769 iterable_batched = iter( 

1770 lambda: tuple(itertools.islice(it, batch_size)), () 

1771 ) 

1772 iterable = ( 

1773 task for batch in iterable_batched for task in batch 

1774 ) 

1775 

1776 # first yield returns None, for internal use only. This ensures 

1777 # that we enter the try/except block and setup the generator. 

1778 yield None 

1779 

1780 # Sequentially call the tasks and yield the results. 

1781 for func, args, kwargs in iterable: 

1782 self.n_dispatched_batches += 1 

1783 self.n_dispatched_tasks += 1 

1784 res = func(*args, **kwargs) 

1785 self.n_completed_tasks += 1 

1786 self.print_progress() 

1787 yield res 

1788 self._nb_consumed += 1 

1789 except BaseException: 

1790 self._exception = True 

1791 self._aborting = True 

1792 self._aborted = True 

1793 raise 

1794 finally: 

1795 self.print_progress() 

1796 self._running = False 

1797 self._iterating = False 

1798 self._original_iterator = None

1810 " Before submitting new tasks, you must wait for the "

1818 self.n_dispatched_batches = 0 

1819 self.n_dispatched_tasks = 0 

1820 self.n_completed_tasks = 0

1837 self._aborted = False

1839 def __call__(self, iterable): 

1840 """Main function to dispatch parallel tasks.""" 

1841 

1842 self._reset_run_tracking() 

1843 self._start_time = time.time() 

1844 

1845 if not self._managed_backend: 

1846 n_jobs = self._initialize_backend() 

1847 else: 

1848 n_jobs = self._effective_n_jobs() 

1849 

1850 if n_jobs == 1: 

1851 # If n_jobs==1, run the computation sequentially and return 

1852 # immediatly to avoid overheads. 

1853 output = self._get_sequential_output(iterable) 

1854 next(output) 

1855 return output if self.return_generator else list(output) 

1856 

1857 # Let's create an ID that uniquely identifies the current call. If the 

1858 # call is interrupted early and that the same instance is immediately 

1859 # re-used, this id will be used to prevent workers that were 

1860 # concurrently finalizing a task from the previous call to run the 

1861 # callback. 

1862 with self._lock:
