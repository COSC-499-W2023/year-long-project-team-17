72 def __getitem__(self, idx): 

73 """ 

74 Return shape at *idx* in sequence, e.g. ``shapes[2]``. 

75 """ 

76 shape_elms = list(self._iter_member_elms()) 

77 try: 

78 shape_elm = shape_elms[idx] 

79 except IndexError: 

80 raise IndexError("shape index out of range") 

81 return self._shape_factory(shape_elm)

83 def __iter__(self): 

84 """ 

85 Generate a reference to each shape in the collection, in sequence. 

86 """ 

87 for shape_elm in self._iter_member_elms(): 

88 yield self._shape_factory(shape_elm)

90 def __len__(self): 

91 """ 

92 Return count of shapes in this shape tree. A group shape contributes 

93 1 to the total, without regard to the number of shapes contained in 

94 the group. 

95 """ 

96 shape_elms = list(self._iter_member_elms()) 

97 return len(shape_elms)

132 @property 

133 def turbo_add_enabled(self): 

134 """True if "turbo-add" mode is enabled. Read/Write. 

135 

136 EXPERIMENTAL: This feature can radically improve performance when 

137 adding large numbers (hundreds of shapes) to a slide. It works by 

138 caching the last shape ID used and incrementing that value to assign 

139 the next shape id. This avoids repeatedly searching all shape ids in 

140 the slide each time a new ID is required. 

141 

142 Performance is not noticeably improved for a slide with a relatively 

143 small number of shapes, but because the search time rises with the 

144 square of the shape count, this option can be useful for optimizing 

145 generation of a slide composed of many shapes. 

146 

147 Shape-id collisions can occur (causing a repair error on load) if 

148 more than one |Slide| object is used to interact with the same slide 

149 in the presentation. Note that the |Slides| collection creates a new 

150 |Slide| object each time a slide is accessed 

151 (e.g. `slide = prs.slides[0]`, so you must be careful to limit use to 

152 a single |Slide| object. 

153 """ 

154 return self._cached_max_shape_id is not None

178 def _next_ph_name(self, ph_type, id, orient): 

179 """ 

180 Next unique placeholder name for placeholder shape of type *ph_type*, 

181 with id number *id* and orientation *orient*. Usually will be standard 

182 placeholder root name suffixed with id-1, e.g. 

183 _next_ph_name(ST_PlaceholderType.TBL, 4, 'horz') ==> 

184 'Table Placeholder 3'. The number is incremented as necessary to make 

185 the name unique within the collection. If *orient* is ``'vert'``, the 

186 placeholder name is prefixed with ``'Vertical '``. 

187 """ 

188 basename = self.ph_basename(ph_type) 

189 

190 # prefix rootname with 'Vertical ' if orient is 'vert' 

191 if orient == ST_Direction.VERT: 

192 basename = "Vertical %s" % basename 

193 

194 # increment numpart as necessary to make name unique 

195 numpart = id - 1 

196 names = self._spTree.xpath("//p:cNvPr/@name") 

197 while True: 

198 name = "%s %d" % (basename, numpart) 

199 if name not in names: 

200 break 

201 numpart += 1 

202 

203 return name

205 @property 

206 def _next_shape_id(self): 

207 """Return a unique shape id suitable for use with a new shape. 

208 

209 The returned id is 1 greater than the maximum shape id used so far. 

210 In practice, the minimum id is 2 because the spTree element is always 

211 assigned id="1". 

212 """ 

213 # ---presence of cached-max-shape-id indicates turbo mode is on--- 

214 if self._cached_max_shape_id is not None: 

215 self._cached_max_shape_id += 1 

216 return self._cached_max_shape_id 

217 

218 return self._spTree.max_shape_id + 1

378 def build_freeform(self, start_x=0, start_y=0, scale=1.0): 

379 """Return |FreeformBuilder| object to specify a freeform shape. 

380 

381 The optional *start_x* and *start_y* arguments specify the starting 

382 pen position in local coordinates. They will be rounded to the 

383 nearest integer before use and each default to zero. 

384 

385 The optional *scale* argument specifies the size of local coordinates 

386 proportional to slide coordinates (EMU). If the vertical scale is 

387 different than the horizontal scale (local coordinate units are 

388 "rectangular"), a pair of numeric values can be provided as the 

389 *scale* argument, e.g. `scale=(1.0, 2.0)`. In this case the first 

390 number is interpreted as the horizontal (X) scale and the second as 

391 the vertical (Y) scale. 

392 

393 A convenient method for calculating scale is to divide a |Length| 

394 object by an equivalent count of local coordinate units, e.g. 

395 `scale = Inches(1)/1000` for 1000 local units per inch. 

396 """ 

397 try: 

398 x_scale, y_scale = scale 

399 except TypeError: 

400 x_scale = y_scale = scale 

401 

402 return FreeformBuilder.new(self, start_x, start_y, x_scale, y_scale)

768class SlidePlaceholders(ParentedElementProxy): 

769 """ 

770 Collection of placeholder shapes on a slide. Supports iteration, 

771 :func:`len`, and dictionary-style lookup on the `idx` value of the 

772 placeholders it contains. 

773 """ 

774 

775 def __getitem__(self, idx): 

776 """ 

777 Access placeholder shape having *idx*. Note that while this looks 

778 like list access, idx is actually a dictionary key and will raise 

779 |KeyError| if no placeholder with that idx value is in the 

780 collection. 

781 """ 

782 for e in self._element.iter_ph_elms(): 

783 if e.ph_idx == idx:

788 """

790 """ 

791 ph_elms = sorted( 

792 [e for e in self._element.iter_ph_elms()], key=lambda e: e.ph_idx 

793 ) 

794 return (SlideShapeFactory(e, self) for e in ph_elms)

796 def __len__(self): 

797 """ 

798 Return count of placeholder shapes. 

799 """ 

800 return len(list(self._element.iter_ph_elms()))
