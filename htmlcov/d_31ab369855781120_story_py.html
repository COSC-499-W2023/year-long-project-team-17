13class BaseStoryPart(XmlPart): 

14 """Base class for story parts. 

15 

16 A story part is one that can contain textual content, such as the document-part and 

17 header or footer parts. These all share content behaviors like `.paragraphs`, 

18 `.add_paragraph()`, `.add_table()` etc. 

19 """ 

20 

21 def get_or_add_image(self, image_descriptor): 

22 """Return (rId, image) pair for image identified by *image_descriptor*. 

23 

24 *rId* is the str key (often like "rId7") for the relationship between this story 

25 part and the image part, reused if already present, newly created if not. 

26 *image* is an |Image| instance providing access to the properties of the image, 

27 such as dimensions and image type. 

28 """ 

29 image_part = self._package.get_or_add_image_part(image_descriptor) 

30 rId = self.relate_to(image_part, RT.IMAGE) 

31 return rId, image_part.image 

32 

33 def get_style(self, style_id, style_type): 

34 """Return the style in this document matching *style_id*. 

35 

36 Returns the default style for *style_type* if *style_id* is |None| or does not 

37 match a defined style of *style_type*. 

38 """ 

39 return self._document_part.get_style(style_id, style_type) 

40 

41 def get_style_id(self, style_or_name, style_type): 

42 """Return str style_id for *style_or_name* of *style_type*. 

43 

44 Returns |None| if the style resolves to the default style for *style_type* or if 

45 *style_or_name* is itself |None|. Raises if *style_or_name* is a style of the 

46 wrong type or names a style not present in the document. 

47 """ 

48 return self._document_part.get_style_id(style_or_name, style_type) 

49 

50 def new_pic_inline(self, image_descriptor, width, height): 

51 """Return a newly-created `w:inline` element. 

52 

53 The element contains the image specified by *image_descriptor* and is scaled 

54 based on the values of *width* and *height*. 

55 """ 

56 rId, image = self.get_or_add_image(image_descriptor) 

57 cx, cy = image.scaled_dimensions(width, height) 

58 shape_id, filename = self.next_id, image.filename 

59 return CT_Inline.new_pic_inline(shape_id, rId, filename, cx, cy)

61 @property 

62 def next_id(self): 

63 """Next available positive integer id value in this story XML document. 

64 

65 The value is determined by incrementing the maximum existing id value. Gaps in 

66 the existing id sequence are not filled. The id attribute value is unique in the 

67 document, without regard to the element type it appears on. 

68 """ 

69 id_str_lst = self._element.xpath('//@id') 

70 used_ids = [int(id_str) for id_str in id_str_lst if id_str.isdigit()] 

71 if not used_ids: 

72 return 1 

73 return max(used_ids) + 1

78 return self.package.main_document_part
