37class NLTKWordTokenizer(TokenizerI): 

38 """ 

39 The NLTK tokenizer that has improved upon the TreebankWordTokenizer. 

40 

41 This is the method that is invoked by ``word_tokenize()``. It assumes that the 

42 text has already been segmented into sentences, e.g. using ``sent_tokenize()``. 

43 

44 The tokenizer is "destructive" such that the regexes applied will munge the 

45 input string to a state beyond re-construction. It is possible to apply 

46 `TreebankWordDetokenizer.detokenize` to the tokenized outputs of 

47 `NLTKDestructiveWordTokenizer.tokenize` but there's no guarantees to 

48 revert to the original string. 

49 """ 

50 

51 # Starting quotes. 

52 STARTING_QUOTES = [ 

53 (re.compile("([«“‘„]|[`]+)", re.U), r" \1 "), 

54 (re.compile(r"^\""), r"``"), 

55 (re.compile(r"(``)"), r" \1 "), 

56 (re.compile(r"([ \(\[{<])(\"|\'{2})"), r"\1 `` "), 

57 (re.compile(r"(?i)(\')(?!re|ve|ll|m|t|s|d|n)(\w)\b", re.U), r"\1 \2"), 

58 ] 

59 

60 # Ending quotes. 

61 ENDING_QUOTES = [ 

62 (re.compile("([»”’])", re.U), r" \1 "), 

63 (re.compile(r"''"), " '' "), 

64 (re.compile(r'"'), " '' "), 

65 (re.compile(r"([^' ])('[sS]|'[mM]|'[dD]|') "), r"\1 \2 "), 

66 (re.compile(r"([^' ])('ll|'LL|'re|'RE|'ve|'VE|n't|N'T) "), r"\1 \2 "), 

67 ] 

68 

69 # For improvements for starting/closing quotes from TreebankWordTokenizer, 

70 # see discussion on https://github.com/nltk/nltk/pull/1437 

71 # Adding to TreebankWordTokenizer, nltk.word_tokenize now splits on 

72 # - chervon quotes u'\xab' and u'\xbb' . 

73 # - unicode quotes u'\u2018', u'\u2019', u'\u201c' and u'\u201d' 

74 # See https://github.com/nltk/nltk/issues/1995#issuecomment-376741608 

75 # Also, behavior of splitting on clitics now follows Stanford CoreNLP 

76 # - clitics covered (?!re|ve|ll|m|t|s|d)(\w)\b 

77 

78 # Punctuation. 

79 PUNCTUATION = [ 

80 (re.compile(r'([^\.])(\.)([\]\)}>"\'' "»”’ " r"]*)\s*$", re.U), r"\1 \2 \3 "), 

81 (re.compile(r"([:,])([^\d])"), r" \1 \2"), 

82 (re.compile(r"([:,])$"), r" \1 "), 

83 ( 

84 re.compile(r"\.{2,}", re.U), 

85 r" \g<0> ", 

86 ), # See https://github.com/nltk/nltk/pull/2322 

87 (re.compile(r"[;@#$%&]"), r" \g<0> "), 

88 ( 

89 re.compile(r'([^\.])(\.)([\]\)}>"\']*)\s*$'), 

90 r"\1 \2\3 ", 

91 ), # Handles the final period. 

92 (re.compile(r"[?!]"), r" \g<0> "), 

93 (re.compile(r"([^'])' "), r"\1 ' "), 

94 ( 

95 re.compile(r"[*]", re.U), 

96 r" \g<0> ", 

97 ), # See https://github.com/nltk/nltk/pull/2322 

98 ] 

99 

100 # Pads parentheses 

101 PARENS_BRACKETS = (re.compile(r"[\]\[\(\)\{\}\<\>]"), r" \g<0> ") 

102 

103 # Optionally: Convert parentheses, brackets and converts them to PTB symbols. 

104 CONVERT_PARENTHESES = [ 

105 (re.compile(r"\("), "-LRB-"), 

106 (re.compile(r"\)"), "-RRB-"), 

107 (re.compile(r"\["), "-LSB-"), 

108 (re.compile(r"\]"), "-RSB-"), 

109 (re.compile(r"\{"), "-LCB-"), 

110 (re.compile(r"\}"), "-RCB-"), 

111 ] 

112 

113 DOUBLE_DASHES = (re.compile(r"--"), r" -- ") 

114 

115 # List of contractions adapted from Robert MacIntyre's tokenizer. 

116 _contractions = MacIntyreContractions() 

117 CONTRACTIONS2 = list(map(re.compile, _contractions.CONTRACTIONS2)) 

118 CONTRACTIONS3 = list(map(re.compile, _contractions.CONTRACTIONS3))

120 def tokenize( 

121 self, text: str, convert_parentheses: bool = False, return_str: bool = False 

122 ) -> List[str]: 

123 r"""Return a tokenized copy of `text`. 

124 

125 >>> from nltk.tokenize import NLTKWordTokenizer 

126 >>> s = '''Good muffins cost $3.88 (roughly 3,36 euros)\nin New York. Please buy me\ntwo of them.\nThanks.''' 

127 >>> NLTKWordTokenizer().tokenize(s) 

128 ['Good', 'muffins', 'cost', '$', '3.88', '(', 'roughly', '3,36', 

129 'euros', ')', 'in', 'New', 'York.', 'Please', 'buy', 'me', 'two', 

130 'of', 'them.', 'Thanks', '.'] 

131 >>> NLTKWordTokenizer().tokenize(s, convert_parentheses=True) 

132 ['Good', 'muffins', 'cost', '$', '3.88', '-LRB-', 'roughly', '3,36', 

133 'euros', '-RRB-', 'in', 'New', 'York.', 'Please', 'buy', 'me', 'two', 

134 'of', 'them.', 'Thanks', '.'] 

135 >>> NLTKWordTokenizer().tokenize(s, return_str=True) 

136 ' Good muffins cost $ 3.88 ( roughly 3,36 euros ) \nin New York. Please buy me\ntwo of them.\nThanks . ' 

137 

138 :param text: A string with a sentence or sentences. 

139 :type text: str 

140 :param convert_parentheses: if True, replace parentheses to PTB symbols, 

141 e.g. `(` to `-LRB-`. Defaults to False. 

142 :type convert_parentheses: bool, optional 

143 :param return_str: If True, return tokens as space-separated string, 

144 defaults to False. 

145 :type return_str: bool, optional 

146 :return: List of tokens from `text`. 

147 :rtype: List[str] 

148 """ 

149 if return_str: 

150 warnings.warn( 

151 "Parameter 'return_str' has been deprecated and should no " 

152 "longer be used.", 

153 category=DeprecationWarning, 

154 stacklevel=2, 

155 ) 

156 

157 for regexp, substitution in self.STARTING_QUOTES: 

158 text = regexp.sub(substitution, text) 

159 

160 for regexp, substitution in self.PUNCTUATION: 

161 text = regexp.sub(substitution, text) 

162 

163 # Handles parentheses. 

164 regexp, substitution = self.PARENS_BRACKETS 

165 text = regexp.sub(substitution, text) 

166 # Optionally convert parentheses 

167 if convert_parentheses: 

168 for regexp, substitution in self.CONVERT_PARENTHESES: 

169 text = regexp.sub(substitution, text) 

170 

171 # Handles double dash. 

172 regexp, substitution = self.DOUBLE_DASHES 

173 text = regexp.sub(substitution, text) 

174 

175 # add extra space to make things easier 

176 text = " " + text + " " 

177 

178 for regexp, substitution in self.ENDING_QUOTES: 

179 text = regexp.sub(substitution, text) 

180 

181 for regexp in self.CONTRACTIONS2: 

182 text = regexp.sub(r" \1 \2 ", text) 

183 for regexp in self.CONTRACTIONS3: 

184 text = regexp.sub(r" \1 \2 ", text) 

185 

186 # We are not using CONTRACTIONS4 since 

187 # they are also commented out in the SED scripts 

188 # for regexp in self._contractions.CONTRACTIONS4: 

189 # text = regexp.sub(r' \1 \2 \3 ', text) 

190 

191 return text.split()

193 def span_tokenize(self, text: str) -> Iterator[Tuple[int, int]]: 

194 r""" 

195 Returns the spans of the tokens in ``text``. 

196 Uses the post-hoc nltk.tokens.align_tokens to return the offset spans. 

197 

198 >>> from nltk.tokenize import NLTKWordTokenizer 

199 >>> s = '''Good muffins cost $3.88\nin New (York). Please (buy) me\ntwo of them.\n(Thanks).''' 

200 >>> expected = [(0, 4), (5, 12), (13, 17), (18, 19), (19, 23), 

201 ... (24, 26), (27, 30), (31, 32), (32, 36), (36, 37), (37, 38), 

202 ... (40, 46), (47, 48), (48, 51), (51, 52), (53, 55), (56, 59), 

203 ... (60, 62), (63, 68), (69, 70), (70, 76), (76, 77), (77, 78)] 

204 >>> list(NLTKWordTokenizer().span_tokenize(s)) == expected 

205 True 

206 >>> expected = ['Good', 'muffins', 'cost', '$', '3.88', 'in', 

207 ... 'New', '(', 'York', ')', '.', 'Please', '(', 'buy', ')', 

208 ... 'me', 'two', 'of', 'them.', '(', 'Thanks', ')', '.'] 

209 >>> [s[start:end] for start, end in NLTKWordTokenizer().span_tokenize(s)] == expected 

210 True 

211 

212 :param text: A string with a sentence or sentences. 

213 :type text: str 

214 :yield: Tuple[int, int] 

215 """ 

216 raw_tokens = self.tokenize(text) 

217 

218 # Convert converted quotes back to original double quotes 

219 # Do this only if original text contains double quote(s) or double 

220 # single-quotes (because '' might be transformed to `` if it is 

221 # treated as starting quotes). 

222 if ('"' in text) or ("''" in text): 

223 # Find double quotes and converted quotes 

224 matched = [m.group() for m in re.finditer(r"``|'{2}|\"", text)] 

225 

226 # Replace converted quotes back to double quotes 

227 tokens = [ 

228 matched.pop(0) if tok in ['"', "``", "''"] else tok 

229 for tok in raw_tokens 

230 ] 

231 else: 

232 tokens = raw_tokens 

233 

234 yield from align_tokens(tokens, text)
