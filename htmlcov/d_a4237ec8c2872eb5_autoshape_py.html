19class Adjustment(object): 

20 """ 

21 An adjustment value for an autoshape. 

22 

23 An adjustment value corresponds to the position of an adjustment handle on 

24 an auto shape. Adjustment handles are the small yellow diamond-shaped 

25 handles that appear on certain auto shapes and allow the outline of the 

26 shape to be adjusted. For example, a rounded rectangle has an adjustment 

27 handle that allows the radius of its corner rounding to be adjusted. 

28 

29 Values are |float| and generally range from 0.0 to 1.0, although the value 

30 can be negative or greater than 1.0 in certain circumstances. 

31 """ 

32 

33 def __init__(self, name, def_val, actual=None): 

34 super(Adjustment, self).__init__() 

35 self.name = name 

36 self.def_val = def_val 

37 self.actual = actual 

38 

39 @property 

40 def effective_value(self): 

41 """ 

42 Read/write |float| representing normalized adjustment value for this 

43 adjustment. Actual values are a large-ish integer expressed in shape 

44 coordinates, nominally between 0 and 100,000. The effective value is 

45 normalized to a corresponding value nominally between 0.0 and 1.0. 

46 Intuitively this represents the proportion of the width or height of 

47 the shape at which the adjustment value is located from its starting 

48 point. For simple shapes such as a rounded rectangle, this intuitive 

49 correspondence holds. For more complicated shapes and at more extreme 

50 shape proportions (e.g. width is much greater than height), the value 

51 can become negative or greater than 1.0. 

52 """ 

53 raw_value = self.actual 

54 if raw_value is None: 

55 raw_value = self.def_val 

56 return self._normalize(raw_value) 

57 

58 @effective_value.setter 

59 def effective_value(self, value): 

60 if not isinstance(value, Number): 

61 tmpl = "adjustment value must be numeric, got '%s'" 

62 raise ValueError(tmpl % value) 

63 self.actual = self._denormalize(value) 

64 

65 @staticmethod 

66 def _denormalize(value): 

67 """ 

68 Return integer corresponding to normalized *raw_value* on unit basis 

69 of 100,000. See Adjustment.normalize for additional details. 

70 """ 

71 return int(value * 100000.0) 

72 

73 @staticmethod 

74 def _normalize(raw_value): 

75 """ 

76 Return normalized value for *raw_value*. A normalized value is a 

77 |float| between 0.0 and 1.0 for nominal raw values between 0 and 

78 100,000. Raw values less than 0 and greater than 100,000 are valid 

79 and return values calculated on the same unit basis of 100,000. 

80 """ 

81 return raw_value / 100000.0 

82 

83 @property 

84 def val(self): 

85 """ 

86 Denormalized effective value (expressed in shape coordinates), 

87 suitable for using in the XML. 

88 """ 

89 return self.actual if self.actual is not None else self.def_val

92class AdjustmentCollection(object): 

93 """ 

94 Sequence of |Adjustment| instances for an auto shape, each representing 

95 an available adjustment for a shape of its type. Supports ``len()`` and 

96 indexed access, e.g. ``shape.adjustments[1] = 0.15``. 

97 """ 

98 

99 def __init__(self, prstGeom): 

100 super(AdjustmentCollection, self).__init__() 

101 self._adjustments_ = self._initialized_adjustments(prstGeom) 

102 self._prstGeom = prstGeom 

103 

104 def __getitem__(self, key): 

105 """Provides indexed access, (e.g. 'adjustments[9]').""" 

106 return self._adjustments_[key].effective_value 

107 

108 def __setitem__(self, key, value): 

109 """ 

110 Provides item assignment via an indexed expression, e.g. 

111 ``adjustments[9] = 999.9``. Causes all adjustment values in 

112 collection to be written to the XML. 

113 """ 

114 self._adjustments_[key].effective_value = value 

115 self._rewrite_guides() 

116 

117 def _initialized_adjustments(self, prstGeom): 

118 """ 

119 Return an initialized list of adjustment values based on the contents 

120 of *prstGeom* 

121 """ 

122 if prstGeom is None: 

123 return [] 

124 davs = AutoShapeType.default_adjustment_values(prstGeom.prst) 

125 adjustments = [Adjustment(name, def_val) for name, def_val in davs] 

126 self._update_adjustments_with_actuals(adjustments, prstGeom.gd_lst) 

127 return adjustments 

128 

129 def _rewrite_guides(self): 

130 """ 

131 Write ``<a:gd>`` elements to the XML, one for each adjustment value. 

132 Any existing guide elements are overwritten. 

133 """ 

134 guides = [(adj.name, adj.val) for adj in self._adjustments_] 

135 self._prstGeom.rewrite_guides(guides) 

136 

137 @staticmethod 

138 def _update_adjustments_with_actuals(adjustments, guides): 

139 """ 

140 Update |Adjustment| instances in *adjustments* with actual values 

141 held in *guides*, a list of ``<a:gd>`` elements. Guides with a name 

142 that does not match an adjustment object are skipped. 

143 """ 

144 adjustments_by_name = dict((adj.name, adj) for adj in adjustments) 

145 for gd in guides: 

146 name = gd.name 

147 actual = int(gd.fmla[4:]) 

148 try: 

149 adjustment = adjustments_by_name[name] 

150 except KeyError: 

151 continue 

152 adjustment.actual = actual 

153 return 

154 

155 @property 

156 def _adjustments(self): 

157 """ 

158 Sequence containing direct references to the |Adjustment| objects 

159 contained in collection. 

160 """ 

161 return tuple(self._adjustments_) 

162 

163 def __len__(self): 

164 """Implement built-in function len()""" 

165 return len(self._adjustments_)

168class AutoShapeType(object): 

169 """ 

170 Return an instance of |AutoShapeType| containing metadata for an auto 

171 shape of type identified by *autoshape_type_id*. Instances are cached, so 

172 no more than one instance for a particular auto shape type is in memory. 

173 

174 Instances provide the following attributes: 

175 

176 .. attribute:: autoshape_type_id 

177 

178 Integer uniquely identifying this auto shape type. Corresponds to a 

179 value in ``pptx.constants.MSO`` like ``MSO_SHAPE.ROUNDED_RECTANGLE``. 

180 

181 .. attribute:: basename 

182 

183 Base part of shape name for auto shapes of this type, e.g. ``Rounded 

184 Rectangle`` becomes ``Rounded Rectangle 99`` when the distinguishing 

185 integer is added to the shape name. 

186 

187 .. attribute:: prst 

188 

189 String identifier for this auto shape type used in the ``<a:prstGeom>`` 

190 element. 

191 

192 .. attribute:: desc 

193 

194 Informal string description of auto shape. 

195 

196 """ 

197 

198 _instances = {} 

199 

200 def __new__(cls, autoshape_type_id): 

201 """ 

202 Only create new instance on first call for content_type. After that, 

203 use cached instance. 

204 """ 

205 # if there's not a matching instance in the cache, create one 

206 if autoshape_type_id not in cls._instances: 

207 inst = super(AutoShapeType, cls).__new__(cls) 

208 cls._instances[autoshape_type_id] = inst 

209 # return the instance; note that __init__() gets called either way 

210 return cls._instances[autoshape_type_id] 

211 

212 def __init__(self, autoshape_type_id): 

213 """Initialize attributes from constant values in pptx.spec""" 

214 # skip loading if this instance is from the cache 

215 if hasattr(self, "_loaded"): 

216 return 

217 # raise on bad autoshape_type_id 

218 if autoshape_type_id not in autoshape_types: 

219 raise KeyError( 

220 "no autoshape type with id '%s' in pptx.spec.autoshape_types" 

221 % autoshape_type_id 

222 ) 

223 # otherwise initialize new instance 

224 autoshape_type = autoshape_types[autoshape_type_id] 

225 self._autoshape_type_id = autoshape_type_id 

226 self._basename = autoshape_type["basename"] 

227 self._loaded = True 

228 

229 @property 

230 def autoshape_type_id(self): 

231 """ 

232 MSO_AUTO_SHAPE_TYPE enumeration value for this auto shape type 

233 """ 

234 return self._autoshape_type_id 

235 

236 @property 

237 def basename(self): 

238 """Base of shape name for this auto shape type. 

239 

240 A shape name is like "Rounded Rectangle 7" and appears as an XML attribute for 

241 example at `p:sp/p:nvSpPr/p:cNvPr{name}`. This basename value is the name less 

242 the distinguishing integer. This value is escaped because at least one 

243 autoshape-type name includes double quotes ('"No" Symbol'). 

244 """ 

245 return saxutils.escape(self._basename, {'"': "&quot;"})

248 def default_adjustment_values(cls, prst): 

249 """ 

250 Return sequence of name, value tuples representing the adjustment 

251 value defaults for the auto shape type identified by *prst*. 

252 """ 

253 return autoshape_types[prst]["avLst"] 

254 

255 @property 

256 def desc(self): 

257 """Informal description of this auto shape type""" 

258 return self._desc 

259 

260 @classmethod 

261 def id_from_prst(cls, prst): 

262 """ 

263 Return auto shape id (e.g. ``MSO_SHAPE.RECTANGLE``) corresponding to 

264 preset geometry keyword *prst*. 

265 """ 

266 return MSO_AUTO_SHAPE_TYPE.from_xml(prst)

268 @property 

269 def prst(self): 

270 """ 

271 Preset geometry identifier string for this auto shape. Used in the 

272 ``prst`` attribute of ``<a:prstGeom>`` element to specify the geometry 

273 to be used in rendering the shape, for example ``'roundRect'``. 

274 """ 

275 return MSO_AUTO_SHAPE_TYPE.to_xml(self._autoshape_type_id)

364 @property 

365 def text(self): 

366 """Read/write. Unicode (str in Python 3) representation of shape text. 

367 

368 The returned string will contain a newline character (``"\\n"``) separating each 

369 paragraph and a vertical-tab (``"\\v"``) character for each line break (soft 

370 carriage return) in the shape's text. 

371 

372 Assignment to *text* replaces all text previously contained in the shape, along 

373 with any paragraph or font formatting applied to it. A newline character 

374 (``"\\n"``) in the assigned text causes a new paragraph to be started. 

375 A vertical-tab (``"\\v"``) character in the assigned text causes a line-break 

376 (soft carriage-return) to be inserted. (The vertical-tab character appears in 

377 clipboard text copied from PowerPoint as its encoding of line-breaks.) 

378 

379 Either bytes (Python 2 str) or unicode (Python 3 str) can be assigned. Bytes can 

380 be 7-bit ASCII or UTF-8 encoded 8-bit bytes. Bytes values are converted to 

381 unicode assuming UTF-8 encoding (which also works for ASCII). 

382 """ 

383 return self.text_frame.text
