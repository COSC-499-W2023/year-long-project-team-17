117 def __init__(self, key: bytes) -> None: 

118 self.S = list(range(256)) 

119 j = 0 

120 for i in range(256): 

121 j = (j + self.S[i] + key[i % len(key)]) % 256 

122 self.S[i], self.S[j] = self.S[j], self.S[i]

124 def encrypt(self, data: bytes) -> bytes: 

125 S = list(self.S) 

126 out = list(0 for _ in range(len(data))) 

127 i, j = 0, 0 

128 for k in range(len(data)): 

129 i = (i + 1) % 256 

130 j = (j + S[i]) % 256 

131 S[i], S[j] = S[j], S[i] 

132 x = S[(S[i] + S[j]) % 256] 

133 out[k] = data[k] ^ x 

134 return bytes(bytearray(out))

238 @staticmethod 

239 def compute_key( 

240 password: bytes, 

241 rev: int, 

242 key_size: int, 

243 o_entry: bytes, 

244 P: int, 

245 id1_entry: bytes, 

246 metadata_encrypted: bool, 

247 ) -> bytes: 

248 """ 

249 Algorithm 2: Computing an encryption key. 

250 

251 a) Pad or truncate the password string to exactly 32 bytes. If the 

252 password string is more than 32 bytes long, 

253 use only its first 32 bytes; if it is less than 32 bytes long, pad it 

254 by appending the required number of 

255 additional bytes from the beginning of the following padding string: 

256 < 28 BF 4E 5E 4E 75 8A 41 64 00 4E 56 FF FA 01 08 

257 2E 2E 00 B6 D0 68 3E 80 2F 0C A9 FE 64 53 69 7A > 

258 That is, if the password string is n bytes long, append 

259 the first 32 - n bytes of the padding string to the end 

260 of the password string. If the password string is empty (zero-length), 

261 meaning there is no user password, 

262 substitute the entire padding string in its place. 

263 

264 b) Initialize the MD5 hash function and pass the result of step (a) 

265 as input to this function. 

266 c) Pass the value of the encryption dictionary’s O entry to the 

267 MD5 hash function. ("Algorithm 3: Computing 

268 the encryption dictionary’s O (owner password) value" shows how the 

269 O value is computed.) 

270 d) Convert the integer value of the P entry to a 32-bit unsigned binary 

271 number and pass these bytes to the 

272 MD5 hash function, low-order byte first. 

273 e) Pass the first element of the file’s file identifier array (the value 

274 of the ID entry in the document’s trailer 

275 dictionary; see Table 15) to the MD5 hash function. 

276 f) (Security handlers of revision 4 or greater) If document metadata is 

277 not being encrypted, pass 4 bytes with 

278 the value 0xFFFFFFFF to the MD5 hash function. 

279 g) Finish the hash. 

280 h) (Security handlers of revision 3 or greater) Do the following 

281 50 times: Take the output from the previous 

282 MD5 hash and pass the first n bytes of the output as input into a new 

283 MD5 hash, where n is the number of 

284 bytes of the encryption key as defined by the value of the encryption 

285 dictionary’s Length entry. 

286 i) Set the encryption key to the first n bytes of the output from the 

287 final MD5 hash, where n shall always be 5 

288 for security handlers of revision 2 but, for security handlers of 

289 revision 3 or greater, shall depend on the 

290 value of the encryption dictionary’s Length entry. 

291 """ 

292 a = _padding(password) 

293 u_hash = hashlib.md5(a) 

294 u_hash.update(o_entry) 

295 u_hash.update(struct.pack("<I", P)) 

296 u_hash.update(id1_entry) 

297 if rev >= 4 and metadata_encrypted is False: 

298 u_hash.update(b"\xff\xff\xff\xff") 

299 u_hash_digest = u_hash.digest() 

300 length = key_size // 8 

301 if rev >= 3: 

302 for _ in range(50): 

303 u_hash_digest = hashlib.md5(u_hash_digest[:length]).digest() 

304 return u_hash_digest[:length]

306 @staticmethod 

307 def compute_O_value_key(owner_password: bytes, rev: int, key_size: int) -> bytes: 

308 """ 

309 Algorithm 3: Computing the encryption dictionary’s O (owner password) value. 

310 

311 a) Pad or truncate the owner password string as described in step (a) 

312 of "Algorithm 2: Computing an encryption key". 

313 If there is no owner password, use the user password instead. 

314 b) Initialize the MD5 hash function and pass the result of step (a) as 

315 input to this function. 

316 c) (Security handlers of revision 3 or greater) Do the following 50 times: 

317 Take the output from the previous 

318 MD5 hash and pass it as input into a new MD5 hash. 

319 d) Create an RC4 encryption key using the first n bytes of the output 

320 from the final MD5 hash, where n shall 

321 always be 5 for security handlers of revision 2 but, for security 

322 handlers of revision 3 or greater, shall 

323 depend on the value of the encryption dictionary’s Length entry. 

324 e) Pad or truncate the user password string as described in step (a) of 

325 "Algorithm 2: Computing an encryption key". 

326 f) Encrypt the result of step (e), using an RC4 encryption function with 

327 the encryption key obtained in step (d). 

328 g) (Security handlers of revision 3 or greater) Do the following 19 times: 

329 Take the output from the previous 

330 invocation of the RC4 function and pass it as input to a new 

331 invocation of the function; use an encryption 

332 key generated by taking each byte of the encryption key obtained in 

333 step (d) and performing an XOR 

334 (exclusive or) operation between that byte and the single-byte value 

335 of the iteration counter (from 1 to 19). 

336 h) Store the output from the final invocation of the RC4 function as 

337 the value of the O entry in the encryption dictionary. 

338 """ 

339 a = _padding(owner_password) 

340 o_hash_digest = hashlib.md5(a).digest() 

341 

342 if rev >= 3: 

343 for _ in range(50): 

344 o_hash_digest = hashlib.md5(o_hash_digest).digest() 

345 

346 rc4_key = o_hash_digest[: key_size // 8] 

347 return rc4_key

402 value of the U entry in the encryption dictionary.

405 u_hash.update(id1_entry) 

406 rc4_enc = RC4_encrypt(key, u_hash.digest()) 

407 for i in range(1, 20): 

408 rc4_key = bytes(bytearray(x ^ i for x in key)) 

409 rc4_enc = RC4_encrypt(rc4_key, rc4_enc) 

410 return _padding(rc4_enc)

495 @staticmethod 

496 def verify_owner_password( 

497 R: int, password: bytes, o_value: bytes, oe_value: bytes, u_value: bytes 

498 ) -> bytes: 

499 """ 

500 Algorithm 3.2a Computing an encryption key. 

501 

502 To understand the algorithm below, it is necessary to treat the O and U strings in the Encrypt dictionary 

503 as made up of three sections. The first 32 bytes are a hash value (explained below). The next 8 bytes are 

504 called the Validation Salt. The final 8 bytes are called the Key Salt. 

505 

506 1. The password string is generated from Unicode input by processing the input string with the SASLprep 

507 (IETF RFC 4013) profile of stringprep (IETF RFC 3454), and then converting to a UTF-8 representation. 

508 2. Truncate the UTF-8 representation to 127 bytes if it is longer than 127 bytes. 

509 3. Test the password against the owner key by computing the SHA-256 hash of the UTF-8 password 

510 concatenated with the 8 bytes of owner Validation Salt, concatenated with the 48-byte U string. If the 

511 32-byte result matches the first 32 bytes of the O string, this is the owner password. 

512 Compute an intermediate owner key by computing the SHA-256 hash of the UTF-8 password 

513 concatenated with the 8 bytes of owner Key Salt, concatenated with the 48-byte U string. The 32-byte 

514 result is the key used to decrypt the 32-byte OE string using AES-256 in CBC mode with no padding and 

515 an initialization vector of zero. The 32-byte result is the file encryption key. 

516 4. Test the password against the user key by computing the SHA-256 hash of the UTF-8 password 

517 concatenated with the 8 bytes of user Validation Salt. If the 32 byte result matches the first 32 bytes of 

518 the U string, this is the user password. 

519 Compute an intermediate user key by computing the SHA-256 hash of the UTF-8 password 

520 concatenated with the 8 bytes of user Key Salt. The 32-byte result is the key used to decrypt the 32-byte 

521 UE string using AES-256 in CBC mode with no padding and an initialization vector of zero. The 32-byte 

522 result is the file encryption key. 

523 5. Decrypt the 16-byte Perms string using AES-256 in ECB mode with an initialization vector of zero and 

524 the file encryption key as the key. Verify that bytes 9-11 of the result are the characters ‘a’, ‘d’, ‘b’. Bytes 

525 0-3 of the decrypted Perms entry, treated as a little-endian integer, are the user permissions. They 

526 should match the value in the P key. 

527 """ 

528 password = password[:127] 

529 if ( 

530 AlgV5.calculate_hash(R, password, o_value[32:40], u_value[:48]) 

531 != o_value[:32] 

532 ): 

533 return b"" 

534 iv = bytes(0 for _ in range(16)) 

535 tmp_key = AlgV5.calculate_hash(R, password, o_value[40:48], u_value[:48]) 

536 key = AES_CBC_decrypt(tmp_key, iv, oe_value) 

537 return key

551 @staticmethod 

552 def calculate_hash(R: int, password: bytes, salt: bytes, udata: bytes) -> bytes: 

553 # from https://github.com/qpdf/qpdf/blob/main/libqpdf/QPDF_encryption.cc 

554 K = hashlib.sha256(password + salt + udata).digest() 

555 if R < 6: 

556 return K 

557 count = 0 

558 while True: 

559 count += 1 

560 K1 = password + K + udata 

561 E = AES_CBC_encrypt(K[:16], K[16:32], K1 * 64) 

562 hash_fn = ( 

563 hashlib.sha256, 

564 hashlib.sha384, 

565 hashlib.sha512, 

566 )[sum(E[:16]) % 3] 

567 K = hash_fn(E).digest() 

568 if count >= 64 and E[-1] <= count - 32: 

569 break 

570 return K[:32]

601 @staticmethod 

602 def compute_U_value(password: bytes, key: bytes) -> Tuple[bytes, bytes]: 

603 """ 

604 Algorithm 3.8 Computing the encryption dictionary’s U (user password) and UE (user encryption key) values 

605 

606 1. Generate 16 random bytes of data using a strong random number generator. The first 8 bytes are the 

607 User Validation Salt. The second 8 bytes are the User Key Salt. Compute the 32-byte SHA-256 hash of 

608 the password concatenated with the User Validation Salt. The 48-byte string consisting of the 32-byte 

609 hash followed by the User Validation Salt followed by the User Key Salt is stored as the U key. 

610 2. Compute the 32-byte SHA-256 hash of the password concatenated with the User Key Salt. Using this 

611 hash as the key, encrypt the file encryption key using AES-256 in CBC mode with no padding and an 

612 initialization vector of zero. The resulting 32-byte string is stored as the UE key. 

613 """ 

614 random_bytes = bytes(random.randrange(0, 256) for _ in range(16)) 

615 val_salt = random_bytes[:8] 

616 key_salt = random_bytes[8:] 

617 u_value = hashlib.sha256(password + val_salt).digest() + val_salt + key_salt 

618 

619 tmp_key = hashlib.sha256(password + key_salt).digest() 

620 iv = bytes(0 for _ in range(16)) 

621 ue_value = AES_CBC_encrypt(tmp_key, iv, key) 

622 return u_value, ue_value 

623 

624 @staticmethod 

625 def compute_O_value( 

626 password: bytes, key: bytes, u_value: bytes 

627 ) -> Tuple[bytes, bytes]: 

628 """ 

629 Algorithm 3.9 Computing the encryption dictionary’s O (owner password) and OE (owner encryption key) values. 

630 

631 1. Generate 16 random bytes of data using a strong random number generator. The first 8 bytes are the 

632 Owner Validation Salt. The second 8 bytes are the Owner Key Salt. Compute the 32-byte SHA-256 hash 

633 of the password concatenated with the Owner Validation Salt and then concatenated with the 48-byte 

634 U string as generated in Algorithm 3.8. The 48-byte string consisting of the 32-byte hash followed by 

635 the Owner Validation Salt followed by the Owner Key Salt is stored as the O key. 

636 2. Compute the 32-byte SHA-256 hash of the password concatenated with the Owner Key Salt and then 

637 concatenated with the 48-byte U string as generated in Algorithm 3.8. Using this hash as the key, 

638 encrypt the file encryption key using AES-256 in CBC mode with no padding and an initialization vector 

639 of zero. The resulting 32-byte string is stored as the OE key. 

640 """ 

641 random_bytes = bytes(random.randrange(0, 256) for _ in range(16)) 

642 val_salt = random_bytes[:8] 

643 key_salt = random_bytes[8:] 

644 o_value = ( 

645 hashlib.sha256(password + val_salt + u_value).digest() + val_salt + key_salt 

646 ) 

647 

648 tmp_key = hashlib.sha256(password + key_salt + u_value).digest() 

649 iv = bytes(0 for _ in range(16)) 

650 oe_value = AES_CBC_encrypt(tmp_key, iv, key) 

651 return o_value, oe_value

653 @staticmethod 

654 def compute_Perms_value(key: bytes, p: int, metadata_encrypted: bool) -> bytes: 

655 """ 

656 Algorithm 3.10 Computing the encryption dictionary’s Perms (permissions) value 

657 

658 1. Extend the permissions (contents of the P integer) to 64 bits by setting the upper 32 bits to all 1’s. (This 

659 allows for future extension without changing the format.) 

660 2. Record the 8 bytes of permission in the bytes 0-7 of the block, low order byte first. 

661 3. Set byte 8 to the ASCII value ' T ' or ' F ' according to the EncryptMetadata Boolean. 

662 4. Set bytes 9-11 to the ASCII characters ' a ', ' d ', ' b '. 

663 5. Set bytes 12-15 to 4 bytes of random data, which will be ignored. 

664 6. Encrypt the 16-byte block using AES-256 in ECB mode with an initialization vector of zero, using the file 

665 encryption key as the key. The result (16 bytes) is stored as the Perms string, and checked for validity 

666 when the file is opened. 

667 """ 

668 b8 = b"T" if metadata_encrypted else b"F" 

669 rr = bytes(random.randrange(0, 256) for _ in range(4)) 

670 data = struct.pack("<I", p) + b"\xff\xff\xff\xff" + b8 + b"adb" + rr 

671 perms = AES_ECB_encrypt(key, data) 

672 return perms

710 def decrypt_object(self, obj: PdfObject, idnum: int, generation: int) -> PdfObject: 

711 """ 

712 Algorithm 1: Encryption of data using the RC4 or AES algorithms. 

713 

714 a) Obtain the object number and generation number from the object identifier of the string or stream to be 

715 encrypted (see 7.3.10, "Indirect Objects"). If the string is a direct object, use the identifier of the indirect 

716 object containing it. 

717 b) For all strings and streams without crypt filter specifier; treating the object number and generation number 

718 as binary integers, extend the original n-byte encryption key to n + 5 bytes by appending the low-order 3 

719 bytes of the object number and the low-order 2 bytes of the generation number in that order, low-order byte 

720 first. (n is 5 unless the value of V in the encryption dictionary is greater than 1, in which case n is the value 

721 of Length divided by 8.) 

722 If using the AES algorithm, extend the encryption key an additional 4 bytes by adding the value “sAlT”, 

723 which corresponds to the hexadecimal values 0x73, 0x41, 0x6C, 0x54. (This addition is done for backward 

724 compatibility and is not intended to provide additional security.) 

725 c) Initialize the MD5 hash function and pass the result of step (b) as input to this function. 

726 d) Use the first (n + 5) bytes, up to a maximum of 16, of the output from the MD5 hash as the key for the RC4 

727 or AES symmetric key algorithms, along with the string or stream data to be encrypted. 

728 If using the AES algorithm, the Cipher Block Chaining (CBC) mode, which requires an initialization vector, 

729 is used. The block size parameter is set to 16 bytes, and the initialization vector is a 16-byte random 

730 number that is stored as the first 16 bytes of the encrypted stream or string. 

731 

732 Algorithm 3.1a Encryption of data using the AES algorithm 

733 1. Use the 32-byte file encryption key for the AES-256 symmetric key algorithm, along with the string or 

734 stream data to be encrypted. 

735 Use the AES algorithm in Cipher Block Chaining (CBC) mode, which requires an initialization vector. The 

736 block size parameter is set to 16 bytes, and the initialization vector is a 16-byte random number that is 

737 stored as the first 16 bytes of the encrypted stream or string. 

738 The output is the encrypted data to be stored in the PDF file. 

739 """ 

740 pack1 = struct.pack("<i", idnum)[:3] 

741 pack2 = struct.pack("<i", generation)[:2] 

742 

743 assert self._key 

744 key = self._key 

745 n = 5 if self.algV == 1 else self.key_size // 8 

746 key_data = key[:n] + pack1 + pack2 

747 key_hash = hashlib.md5(key_data) 

748 rc4_key = key_hash.digest()[: min(n + 5, 16)] 

749 # for AES-128 

750 key_hash.update(b"sAlT") 

751 aes128_key = key_hash.digest()[: min(n + 5, 16)] 

752 

753 # for AES-256 

754 aes256_key = key 

755 

756 stmCrypt = self._get_crypt(self.StmF, rc4_key, aes128_key, aes256_key) 

757 StrCrypt = self._get_crypt(self.StrF, rc4_key, aes128_key, aes256_key) 

758 efCrypt = self._get_crypt(self.EFF, rc4_key, aes128_key, aes256_key) 

759 

760 cf = CryptFilter(stmCrypt, StrCrypt, efCrypt) 

761 return cf.decrypt_object(obj)

764 def _get_crypt( 

765 method: str, rc4_key: bytes, aes128_key: bytes, aes256_key: bytes 

766 ) -> CryptBase: 

767 if method == "/AESV3": 

768 return CryptAES(aes256_key)

776 def verify(self, password: Union[bytes, str]) -> PasswordType: 

777 if isinstance(password, str): 

778 try: 

779 pwd = password.encode("latin-1") 

780 except Exception: # noqa 

781 pwd = password.encode("utf-8") 

782 else: 

783 pwd = password 

784 

785 key, rc = self.verify_v4(pwd) if self.algV <= 4 else self.verify_v5(pwd) 

786 if rc != PasswordType.NOT_DECRYPTED: 

787 self._password_type = rc 

788 self._key = key 

789 return rc

791 def verify_v4(self, password: bytes) -> Tuple[bytes, PasswordType]: 

792 R = cast(int, self.entry["/R"]) 

793 P = cast(int, self.entry["/P"]) 

794 P = (P + 0x100000000) % 0x100000000 # maybe < 0 

795 # make type(metadata_encrypted) == bool 

796 em = self.entry.get("/EncryptMetadata") 

797 metadata_encrypted = em.value if em else True 

798 o_entry = cast(ByteStringObject, self.entry["/O"].get_object()).original_bytes 

799 u_entry = cast(ByteStringObject, self.entry["/U"].get_object()).original_bytes 

800 

801 # verify owner password first 

802 key = AlgV4.verify_owner_password( 

803 password, 

804 R, 

805 self.key_size, 

806 o_entry, 

807 u_entry, 

808 P, 

809 self.id1_entry, 

810 metadata_encrypted, 

811 ) 

812 if key: 

813 return key, PasswordType.OWNER_PASSWORD 

814 key = AlgV4.verify_user_password( 

815 password, 

816 R, 

817 self.key_size, 

818 o_entry, 

819 u_entry, 

820 P, 

821 self.id1_entry, 

822 metadata_encrypted, 

823 ) 

824 if key: 

825 return key, PasswordType.USER_PASSWORD 

826 return b"", PasswordType.NOT_DECRYPTED

828 def verify_v5(self, password: bytes) -> Tuple[bytes, PasswordType]: 

829 # TODO: use SASLprep process 

830 o_entry = cast(ByteStringObject, self.entry["/O"].get_object()).original_bytes 

831 u_entry = cast(ByteStringObject, self.entry["/U"].get_object()).original_bytes 

832 oe_entry = cast(ByteStringObject, self.entry["/OE"].get_object()).original_bytes 

833 ue_entry = cast(ByteStringObject, self.entry["/UE"].get_object()).original_bytes 

834 

835 # verify owner password first 

836 key = AlgV5.verify_owner_password( 

837 self.algR, password, o_entry, oe_entry, u_entry 

838 ) 

839 rc = PasswordType.OWNER_PASSWORD 

840 if not key: 

841 key = AlgV5.verify_user_password(self.algR, password, u_entry, ue_entry) 

842 rc = PasswordType.USER_PASSWORD 

843 if not key: 

844 return b"", PasswordType.NOT_DECRYPTED 

845 

846 # verify Perms 

847 perms = cast(ByteStringObject, self.entry["/Perms"].get_object()).original_bytes 

848 P = cast(int, self.entry["/P"]) 

849 P = (P + 0x100000000) % 0x100000000 # maybe < 0 

850 metadata_encrypted = self.entry.get("/EncryptMetadata", True) 

851 if not AlgV5.verify_perms(key, perms, P, metadata_encrypted): 

852 logger_warning("ignore '/Perms' verify failed", __name__) 

853 return key, rc

855 @staticmethod 

856 def read(encryption_entry: DictionaryObject, first_id_entry: bytes) -> "Encryption": 

857 filter = encryption_entry.get("/Filter") 

858 if filter != "/Standard": 

859 raise NotImplementedError( 

860 "only Standard PDF encryption handler is available" 

861 ) 

862 if "/SubFilter" in encryption_entry: 

863 raise NotImplementedError("/SubFilter NOT supported") 

864 

865 StmF = "/V2" 

866 StrF = "/V2" 

867 EFF = "/V2" 

868 

869 V = encryption_entry.get("/V", 0) 

870 if V not in (1, 2, 3, 4, 5): 

871 raise NotImplementedError(f"Encryption V={V} NOT supported") 

872 if V >= 4: 

873 filters = encryption_entry["/CF"] 

874 

875 StmF = encryption_entry.get("/StmF", "/Identity") 

876 StrF = encryption_entry.get("/StrF", "/Identity") 

877 EFF = encryption_entry.get("/EFF", StmF) 

878 

879 if StmF != "/Identity": 

880 StmF = filters[StmF]["/CFM"] # type: ignore 

881 if StrF != "/Identity": 

882 StrF = filters[StrF]["/CFM"] # type: ignore 

883 if EFF != "/Identity": 

884 EFF = filters[EFF]["/CFM"] # type: ignore 

885 

886 allowed_methods = ("/Identity", "/V2", "/AESV2", "/AESV3") 

887 if StmF not in allowed_methods: 

888 raise NotImplementedError("StmF Method {StmF} NOT supported!") 

889 if StrF not in allowed_methods: 

890 raise NotImplementedError(f"StrF Method {StrF} NOT supported!") 

891 if EFF not in allowed_methods: 

892 raise NotImplementedError(f"EFF Method {EFF} NOT supported!") 

893 

894 R = cast(int, encryption_entry["/R"]) 

895 return Encryption(V, R, encryption_entry, first_id_entry, StmF, StrF, EFF)
