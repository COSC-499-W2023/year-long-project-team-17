16class Video(object): 

17 """Immutable value object representing a video such as MP4.""" 

18 

19 def __init__(self, blob, mime_type, filename): 

20 super(Video, self).__init__() 

21 self._blob = blob 

22 self._mime_type = mime_type 

23 self._filename = filename 

24 

25 @classmethod 

26 def from_blob(cls, blob, mime_type, filename=None): 

27 """Return a new |Video| object loaded from image binary in *blob*.""" 

28 return cls(blob, mime_type, filename) 

29 

30 @classmethod 

31 def from_path_or_file_like(cls, movie_file, mime_type): 

32 """Return a new |Video| object containing video in *movie_file*. 

33 

34 *movie_file* can be either a path (string) or a file-like 

35 (e.g. StringIO) object. 

36 """ 

37 if is_string(movie_file): 

38 # treat movie_file as a path 

39 with open(movie_file, "rb") as f: 

40 blob = f.read() 

41 filename = os.path.basename(movie_file) 

42 else: 

43 # assume movie_file is a file-like object 

44 blob = movie_file.read() 

45 filename = None 

46 

47 return cls.from_blob(blob, mime_type, filename) 

48 

49 @property 

50 def blob(self): 

51 """The bytestream of the media "file".""" 

52 return self._blob 

53 

54 @property 

55 def content_type(self): 

56 """MIME-type of this media, e.g. `'video/mp4'`.""" 

57 return self._mime_type 

58 

59 @property 

60 def ext(self): 

61 """Return the file extension for this video, e.g. 'mp4'. 

62 

63 The extension is that from the actual filename if known. Otherwise 

64 it is the lowercase canonical extension for the video's MIME type. 

65 'vid' is used if the MIME type is 'video/unknown'. 

66 """ 

67 if self._filename: 

68 return os.path.splitext(self._filename)[1].lstrip(".") 

69 return { 

70 CT.ASF: "asf", 

71 CT.AVI: "avi", 

72 CT.MOV: "mov", 

73 CT.MP4: "mp4", 

74 CT.MPG: "mpg", 

75 CT.MS_VIDEO: "avi", 

76 CT.SWF: "swf", 

77 CT.WMV: "wmv", 

78 CT.X_MS_VIDEO: "avi", 

79 }.get(self._mime_type, "vid") 

80 

81 @property 

82 def filename(self): 

83 """Return a filename.ext string appropriate to this video. 

84 

85 The base filename from the original path is used if this image was 

86 loaded from the filesystem. If no filename is available, such as when 

87 the video object is created from an in-memory stream, the string 

88 'movie.{ext}' is used where 'ext' is suitable to the video format, 

89 such as 'mp4'. 

90 """ 

91 if self._filename is not None: 

92 return self._filename 

93 return "movie.%s" % self.ext 

94 

95 @lazyproperty 

96 def sha1(self): 

97 """The SHA1 hash digest for the binary "file" of this video. 

98 

99 Example: `'1be010ea47803b00e140b852765cdf84f491da47'` 

100 """ 

101 return hashlib.sha1(self._blob).hexdigest()
