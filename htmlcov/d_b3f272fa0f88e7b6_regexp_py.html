19class ChunkString: 

20 """ 

21 A string-based encoding of a particular chunking of a text. 

22 Internally, the ``ChunkString`` class uses a single string to 

23 encode the chunking of the input text. This string contains a 

24 sequence of angle-bracket delimited tags, with chunking indicated 

25 by braces. An example of this encoding is:: 

26 

27 {<DT><JJ><NN>}<VBN><IN>{<DT><NN>}<.>{<DT><NN>}<VBD><.> 

28 

29 ``ChunkString`` are created from tagged texts (i.e., lists of 

30 ``tokens`` whose type is ``TaggedType``). Initially, nothing is 

31 chunked. 

32 

33 The chunking of a ``ChunkString`` can be modified with the ``xform()`` 

34 method, which uses a regular expression to transform the string 

35 representation. These transformations should only add and remove 

36 braces; they should *not* modify the sequence of angle-bracket 

37 delimited tags. 

38 

39 :type _str: str 

40 :ivar _str: The internal string representation of the text's 

41 encoding. This string representation contains a sequence of 

42 angle-bracket delimited tags, with chunking indicated by 

43 braces. An example of this encoding is:: 

44 

45 {<DT><JJ><NN>}<VBN><IN>{<DT><NN>}<.>{<DT><NN>}<VBD><.> 

46 

47 :type _pieces: list(tagged tokens and chunks) 

48 :ivar _pieces: The tagged tokens and chunks encoded by this ``ChunkString``. 

49 :ivar _debug: The debug level. See the constructor docs. 

50 

51 :cvar IN_CHUNK_PATTERN: A zero-width regexp pattern string that 

52 will only match positions that are in chunks. 

53 :cvar IN_STRIP_PATTERN: A zero-width regexp pattern string that 

54 will only match positions that are in strips. 

55 """ 

56 

57 CHUNK_TAG_CHAR = r"[^\{\}<>]" 

58 CHUNK_TAG = r"(<%s+?>)" % CHUNK_TAG_CHAR 

59 

60 IN_CHUNK_PATTERN = r"(?=[^\{]*\})" 

61 IN_STRIP_PATTERN = r"(?=[^\}]*(\{|$))" 

62 

63 # These are used by _verify 

64 _CHUNK = r"(\{%s+?\})+?" % CHUNK_TAG 

65 _STRIP = r"(%s+?)+?" % CHUNK_TAG 

66 _VALID = re.compile(r"^(\{?%s\}?)*?$" % CHUNK_TAG) 

67 _BRACKETS = re.compile(r"[^\{\}]+") 

68 _BALANCED_BRACKETS = re.compile(r"(\{\})*$") 

69 

70 def __init__(self, chunk_struct, debug_level=1): 

71 """ 

72 Construct a new ``ChunkString`` that encodes the chunking of 

73 the text ``tagged_tokens``. 

74 

75 :type chunk_struct: Tree 

76 :param chunk_struct: The chunk structure to be further chunked. 

77 :type debug_level: int 

78 :param debug_level: The level of debugging which should be 

79 applied to transformations on the ``ChunkString``. The 

80 valid levels are: 

81 

82 - 0: no checks 

83 - 1: full check on to_chunkstruct 

84 - 2: full check on to_chunkstruct and cursory check after 

85 each transformation. 

86 - 3: full check on to_chunkstruct and full check after 

87 each transformation. 

88 

89 We recommend you use at least level 1. You should 

90 probably use level 3 if you use any non-standard 

91 subclasses of ``RegexpChunkRule``. 

92 """ 

93 self._root_label = chunk_struct.label() 

94 self._pieces = chunk_struct[:] 

95 tags = [self._tag(tok) for tok in self._pieces] 

96 self._str = "<" + "><".join(tags) + ">" 

97 self._debug = debug_level 

98 

99 def _tag(self, tok): 

100 if isinstance(tok, tuple): 

101 return tok[1] 

102 elif isinstance(tok, Tree): 

103 return tok.label() 

104 else: 

105 raise ValueError("chunk structures must contain tagged " "tokens or trees") 

106 

107 def _verify(self, s, verify_tags): 

108 """ 

109 Check to make sure that ``s`` still corresponds to some chunked 

110 version of ``_pieces``. 

111 

112 :type verify_tags: bool 

113 :param verify_tags: Whether the individual tags should be 

114 checked. If this is false, ``_verify`` will check to make 

115 sure that ``_str`` encodes a chunked version of *some* 

116 list of tokens. If this is true, then ``_verify`` will 

117 check to make sure that the tags in ``_str`` match those in 

118 ``_pieces``. 

119 

120 :raise ValueError: if the internal string representation of 

121 this ``ChunkString`` is invalid or not consistent with _pieces. 

122 """ 

123 # Check overall form 

124 if not ChunkString._VALID.match(s): 

125 raise ValueError( 

126 "Transformation generated invalid " "chunkstring:\n %s" % s 

127 ) 

128 

129 # Check that parens are balanced. If the string is long, we 

130 # have to do this in pieces, to avoid a maximum recursion 

131 # depth limit for regular expressions. 

132 brackets = ChunkString._BRACKETS.sub("", s) 

133 for i in range(1 + len(brackets) // 5000): 

134 substr = brackets[i * 5000 : i * 5000 + 5000] 

135 if not ChunkString._BALANCED_BRACKETS.match(substr): 

136 raise ValueError( 

137 "Transformation generated invalid " "chunkstring:\n %s" % s 

138 ) 

139 

140 if verify_tags <= 0: 

141 return 

142 

143 tags1 = (re.split(r"[\{\}<>]+", s))[1:-1] 

144 tags2 = [self._tag(piece) for piece in self._pieces] 

145 if tags1 != tags2: 

146 raise ValueError( 

147 "Transformation generated invalid " "chunkstring: tag changed" 

148 ) 

149 

150 def to_chunkstruct(self, chunk_label="CHUNK"): 

151 """ 

152 Return the chunk structure encoded by this ``ChunkString``. 

153 

154 :rtype: Tree 

155 :raise ValueError: If a transformation has generated an 

156 invalid chunkstring. 

157 """ 

158 if self._debug > 0: 

159 self._verify(self._str, 1) 

160 

161 # Use this alternating list to create the chunkstruct. 

162 pieces = [] 

163 index = 0 

164 piece_in_chunk = 0 

165 for piece in re.split("[{}]", self._str): 

166 

167 # Find the list of tokens contained in this piece. 

168 length = piece.count("<") 

169 subsequence = self._pieces[index : index + length] 

170 

171 # Add this list of tokens to our pieces. 

172 if piece_in_chunk: 

173 pieces.append(Tree(chunk_label, subsequence)) 

174 else: 

175 pieces += subsequence 

176 

177 # Update index, piece_in_chunk 

178 index += length 

179 piece_in_chunk = not piece_in_chunk 

180 

181 return Tree(self._root_label, pieces) 

182 

183 def xform(self, regexp, repl): 

184 """ 

185 Apply the given transformation to the string encoding of this 

186 ``ChunkString``. In particular, find all occurrences that match 

187 ``regexp``, and replace them using ``repl`` (as done by 

188 ``re.sub``). 

189 

190 This transformation should only add and remove braces; it 

191 should *not* modify the sequence of angle-bracket delimited 

192 tags. Furthermore, this transformation may not result in 

193 improper bracketing. Note, in particular, that bracketing may 

194 not be nested. 

195 

196 :type regexp: str or regexp 

197 :param regexp: A regular expression matching the substring 

198 that should be replaced. This will typically include a 

199 named group, which can be used by ``repl``. 

200 :type repl: str 

201 :param repl: An expression specifying what should replace the 

202 matched substring. Typically, this will include a named 

203 replacement group, specified by ``regexp``. 

204 :rtype: None 

205 :raise ValueError: If this transformation generated an 

206 invalid chunkstring. 

207 """ 

208 # Do the actual substitution 

209 s = re.sub(regexp, repl, self._str) 

210 

211 # The substitution might have generated "empty chunks" 

212 # (substrings of the form "{}"). Remove them, so they don't 

213 # interfere with other transformations. 

214 s = re.sub(r"\{\}", "", s) 

215 

216 # Make sure that the transformation was legal. 

217 if self._debug > 1: 

218 self._verify(s, self._debug - 2) 

219 

220 # Commit the transformation. 

221 self._str = s 

222 

223 def __repr__(self): 

224 """ 

225 Return a string representation of this ``ChunkString``. 

226 It has the form:: 

227 

228 <ChunkString: '{<DT><JJ><NN>}<VBN><IN>{<DT><NN>}'> 

229 

230 :rtype: str 

231 """ 

232 return "<ChunkString: %s>" % repr(self._str) 

233 

234 def __str__(self): 

235 """ 

236 Return a formatted representation of this ``ChunkString``. 

237 This representation will include extra spaces to ensure that 

238 tags will line up with the representation of other 

239 ``ChunkStrings`` for the same text, regardless of the chunking. 

240 

241 :rtype: str 

242 """ 

243 # Add spaces to make everything line up. 

244 str = re.sub(r">(?!\})", r"> ", self._str) 

245 str = re.sub(r"([^\{])<", r"\1 <", str) 

246 if str[0] == "<": 

247 str = " " + str 

248 return str

256class RegexpChunkRule: 

257 """ 

258 A rule specifying how to modify the chunking in a ``ChunkString``, 

259 using a transformational regular expression. The 

260 ``RegexpChunkRule`` class itself can be used to implement any 

261 transformational rule based on regular expressions. There are 

262 also a number of subclasses, which can be used to implement 

263 simpler types of rules, based on matching regular expressions. 

264 

265 Each ``RegexpChunkRule`` has a regular expression and a 

266 replacement expression. When a ``RegexpChunkRule`` is "applied" 

267 to a ``ChunkString``, it searches the ``ChunkString`` for any 

268 substring that matches the regular expression, and replaces it 

269 using the replacement expression. This search/replace operation 

270 has the same semantics as ``re.sub``. 

271 

272 Each ``RegexpChunkRule`` also has a description string, which 

273 gives a short (typically less than 75 characters) description of 

274 the purpose of the rule. 

275 

276 This transformation defined by this ``RegexpChunkRule`` should 

277 only add and remove braces; it should *not* modify the sequence 

278 of angle-bracket delimited tags. Furthermore, this transformation 

279 may not result in nested or mismatched bracketing. 

280 """ 

281 

282 def __init__(self, regexp, repl, descr): 

283 """ 

284 Construct a new RegexpChunkRule. 

285 

286 :type regexp: regexp or str 

287 :param regexp: The regular expression for this ``RegexpChunkRule``. 

288 When this rule is applied to a ``ChunkString``, any 

289 substring that matches ``regexp`` will be replaced using 

290 the replacement string ``repl``. Note that this must be a 

291 normal regular expression, not a tag pattern. 

292 :type repl: str 

293 :param repl: The replacement expression for this ``RegexpChunkRule``. 

294 When this rule is applied to a ``ChunkString``, any substring 

295 that matches ``regexp`` will be replaced using ``repl``. 

296 :type descr: str 

297 :param descr: A short description of the purpose and/or effect 

298 of this rule. 

299 """ 

300 if isinstance(regexp, str): 

301 regexp = re.compile(regexp) 

302 self._repl = repl 

303 self._descr = descr 

304 self._regexp = regexp 

305

308 """ 

309 Apply this rule to the given ``ChunkString``. See the 

310 class reference documentation for a description of what it 

311 means to apply a rule. 

312 

313 :type chunkstr: ChunkString 

314 :param chunkstr: The chunkstring to which this rule is applied. 

315 :rtype: None 

316 :raise ValueError: If this transformation generated an 

317 invalid chunkstring. 

318 """ 

319 chunkstr.xform(self._regexp, self._repl) 

320

322 """ 

323 Return a short description of the purpose and/or effect of 

324 this rule. 

325 

326 :rtype: str 

327 """ 

328 return self._descr

350 @staticmethod 

351 def fromstring(s): 

352 """ 

353 Create a RegexpChunkRule from a string description. 

354 Currently, the following formats are supported:: 

355 

356 {regexp} # chunk rule 

357 }regexp{ # strip rule 

358 regexp}{regexp # split rule 

359 regexp{}regexp # merge rule 

360 

361 Where ``regexp`` is a regular expression for the rule. Any 

362 text following the comment marker (``#``) will be used as 

363 the rule's description: 

364 

365 >>> from nltk.chunk.regexp import RegexpChunkRule 

366 >>> RegexpChunkRule.fromstring('{<DT>?<NN.*>+}') 

367 <ChunkRule: '<DT>?<NN.*>+'> 

368 """ 

369 # Split off the comment (but don't split on '\#') 

370 m = re.match(r"(?P<rule>(\\.|[^#])*)(?P<comment>#.*)?", s) 

371 rule = m.group("rule").strip() 

372 comment = (m.group("comment") or "")[1:].strip() 

373 

374 # Pattern bodies: chunk, strip, split, merge 

375 try: 

376 if not rule: 

377 raise ValueError("Empty chunk pattern") 

378 if rule[0] == "{" and rule[-1] == "}": 

379 return ChunkRule(rule[1:-1], comment) 

380 elif rule[0] == "}" and rule[-1] == "{": 

381 return StripRule(rule[1:-1], comment) 

382 elif "}{" in rule: 

383 left, right = rule.split("}{") 

384 return SplitRule(left, right, comment) 

385 elif "{}" in rule: 

386 left, right = rule.split("{}") 

387 return MergeRule(left, right, comment) 

388 elif re.match("[^{}]*{[^{}]*}[^{}]*", rule): 

389 left, chunk, right = re.split("[{}]", rule) 

390 return ChunkRuleWithContext(left, chunk, right, comment) 

391 else: 

392 raise ValueError("Illegal chunk pattern: %s" % rule) 

393 except (ValueError, re.error) as e: 

394 raise ValueError("Illegal chunk pattern: %s" % rule) from e

527 """ 

528 A rule specifying how to merge chunks in a ``ChunkString``, using

530 When applied to a ``ChunkString``, it will find any chunk whose end 

531 matches left pattern, and immediately followed by a chunk whose 

532 beginning matches right pattern. It will then merge those two 

533 chunks into a single chunk. 

534 """ 

535 

536 def __init__(self, left_tag_pattern, right_tag_pattern, descr): 

537 """ 

538 Construct a new ``MergeRule``. 

539 

540 :type right_tag_pattern: str 

541 :param right_tag_pattern: This rule's right tag 

542 pattern. When applied to a ``ChunkString``, this 

543 rule will find any chunk whose end matches 

544 ``left_tag_pattern``, and immediately followed by a chunk 

545 whose beginning matches this pattern. It will 

546 then merge those two chunks into a single chunk. 

547 :type left_tag_pattern: str 

548 :param left_tag_pattern: This rule's left tag 

549 pattern. When applied to a ``ChunkString``, this 

550 rule will find any chunk whose end matches 

551 this pattern, and immediately followed by a chunk 

552 whose beginning matches ``right_tag_pattern``. It will 

553 then merge those two chunks into a single chunk.

555 :type descr: str 

556 :param descr: A short description of the purpose and/or effect

669 When applied to a ``ChunkString``, it will find any chunk whose beginning

671 end matches left pattern. It will then expand the chunk to incorporate 

672 the new material on the left. 

673 """

675 def __init__(self, left_tag_pattern, right_tag_pattern, descr): 

676 """ 

677 Construct a new ``ExpandRightRule``. 

678 

679 :type right_tag_pattern: str 

680 :param right_tag_pattern: This rule's right tag 

681 pattern. When applied to a ``ChunkString``, this 

682 rule will find any chunk whose beginning matches 

683 ``right_tag_pattern``, and immediately preceded by a strip 

684 whose end matches this pattern. It will 

685 then merge those two chunks into a single chunk. 

686 :type left_tag_pattern: str 

687 :param left_tag_pattern: This rule's left tag 

688 pattern. When applied to a ``ChunkString``, this 

689 rule will find any chunk whose beginning matches 

690 this pattern, and immediately preceded by a strip 

691 whose end matches ``left_tag_pattern``. It will 

692 then expand the chunk to incorporate the new material on the left. 

693 

694 :type descr: str 

695 :param descr: A short description of the purpose and/or effect

697 """ 

698 # Ensure that the individual patterns are coherent. E.g., if 

699 # left='(' and right=')', then this will raise an exception: 

700 re.compile(tag_pattern2re_pattern(left_tag_pattern)) 

701 re.compile(tag_pattern2re_pattern(right_tag_pattern)) 

702 

703 self._left_tag_pattern = left_tag_pattern 

704 self._right_tag_pattern = right_tag_pattern 

705 regexp = re.compile( 

706 r"(?P<left>%s)\{(?P<right>%s)" 

707 % ( 

708 tag_pattern2re_pattern(left_tag_pattern), 

709 tag_pattern2re_pattern(right_tag_pattern), 

710 ) 

711 )

893CHUNK_TAG_PATTERN = re.compile( 

894 r"^(({}|<{}>)*)$".format(r"([^\{\}<>]|\{\d+,?\}|\{\d*,\d+\})+", r"[^\{\}<>]+")

898def tag_pattern2re_pattern(tag_pattern): 

899 """ 

900 Convert a tag pattern to a regular expression pattern. A "tag 

901 pattern" is a modified version of a regular expression, designed 

902 for matching sequences of tags. The differences between regular 

903 expression patterns and tag patterns are: 

904 

905 - In tag patterns, ``'<'`` and ``'>'`` act as parentheses; so 

906 ``'<NN>+'`` matches one or more repetitions of ``'<NN>'``, not 

907 ``'<NN'`` followed by one or more repetitions of ``'>'``. 

908 - Whitespace in tag patterns is ignored. So 

909 ``'<DT> | <NN>'`` is equivalent to ``'<DT>|<NN>'`` 

910 - In tag patterns, ``'.'`` is equivalent to ``'[^{}<>]'``; so 

911 ``'<NN.*>'`` matches any single tag starting with ``'NN'``. 

912 

913 In particular, ``tag_pattern2re_pattern`` performs the following 

914 transformations on the given pattern: 

915 

916 - Replace '.' with '[^<>{}]' 

917 - Remove any whitespace 

918 - Add extra parens around '<' and '>', to make '<' and '>' act 

919 like parentheses. E.g., so that in '<NN>+', the '+' has scope 

920 over the entire '<NN>'; and so that in '<NN|IN>', the '|' has 

921 scope over 'NN' and 'IN', but not '<' or '>'. 

922 - Check to make sure the resulting pattern is valid. 

923 

924 :type tag_pattern: str 

925 :param tag_pattern: The tag pattern to convert to a regular 

926 expression pattern. 

927 :raise ValueError: If ``tag_pattern`` is not a valid tag pattern. 

928 In particular, ``tag_pattern`` should not include braces; and it 

929 should not contain nested or mismatched angle-brackets. 

930 :rtype: str 

931 :return: A regular expression pattern corresponding to 

932 ``tag_pattern``. 

933 """ 

934 # Clean up the regular expression 

935 tag_pattern = re.sub(r"\s", "", tag_pattern) 

936 tag_pattern = re.sub(r"<", "(<(", tag_pattern) 

937 tag_pattern = re.sub(r">", ")>)", tag_pattern) 

938 

939 # Check the regular expression 

940 if not CHUNK_TAG_PATTERN.match(tag_pattern): 

941 raise ValueError("Bad tag pattern: %r" % tag_pattern) 

942 

943 # Replace "." with CHUNK_TAG_CHAR. 

944 # We have to do this after, since it adds {}[]<>s, which would 

945 # confuse CHUNK_TAG_PATTERN. 

946 # PRE doesn't have lookback assertions, so reverse twice, and do 

947 # the pattern backwards (with lookahead assertions). This can be 

948 # made much cleaner once we can switch back to SRE. 

949 def reverse_str(str): 

950 lst = list(str) 

951 lst.reverse() 

952 return "".join(lst) 

953 

954 tc_rev = reverse_str(ChunkString.CHUNK_TAG_CHAR) 

955 reversed = reverse_str(tag_pattern) 

956 reversed = re.sub(r"\.(?!\\(\\\\)*($|[^\\]))", tc_rev, reversed) 

957 tag_pattern = reverse_str(reversed) 

958 

959 return tag_pattern

967class RegexpChunkParser(ChunkParserI): 

968 """ 

969 A regular expression based chunk parser. ``RegexpChunkParser`` uses a 

970 sequence of "rules" to find chunks of a single type within a 

971 text. The chunking of the text is encoded using a ``ChunkString``, 

972 and each rule acts by modifying the chunking in the 

973 ``ChunkString``. The rules are all implemented using regular 

974 expression matching and substitution. 

975 

976 The ``RegexpChunkRule`` class and its subclasses (``ChunkRule``, 

977 ``StripRule``, ``UnChunkRule``, ``MergeRule``, and ``SplitRule``) 

978 define the rules that are used by ``RegexpChunkParser``. Each rule 

979 defines an ``apply()`` method, which modifies the chunking encoded 

980 by a given ``ChunkString``. 

981 

982 :type _rules: list(RegexpChunkRule) 

983 :ivar _rules: The list of rules that should be applied to a text. 

984 :type _trace: int 

985 :ivar _trace: The default level of tracing. 

986 

987 """ 

988 

989 def __init__(self, rules, chunk_label="NP", root_label="S", trace=0): 

990 """ 

991 Construct a new ``RegexpChunkParser``. 

992 

993 :type rules: list(RegexpChunkRule) 

994 :param rules: The sequence of rules that should be used to 

995 generate the chunking for a tagged text. 

996 :type chunk_label: str 

997 :param chunk_label: The node value that should be used for 

998 chunk subtrees. This is typically a short string 

999 describing the type of information contained by the chunk, 

1000 such as ``"NP"`` for base noun phrases. 

1001 :type root_label: str 

1002 :param root_label: The node value that should be used for the 

1003 top node of the chunk structure. 

1004 :type trace: int 

1005 :param trace: The level of tracing that should be used when 

1006 parsing a text. ``0`` will generate no tracing output; 

1007 ``1`` will generate normal tracing output; and ``2`` or 

1008 higher will generate verbose tracing output. 

1009 """ 

1010 self._rules = rules 

1011 self._trace = trace 

1012 self._chunk_label = chunk_label 

1013 self._root_label = root_label 

1014 

1015 def _trace_apply(self, chunkstr, verbose): 

1016 """ 

1017 Apply each rule of this ``RegexpChunkParser`` to ``chunkstr``, in 

1018 turn. Generate trace output between each rule. If ``verbose`` 

1019 is true, then generate verbose output. 

1020 

1021 :type chunkstr: ChunkString 

1022 :param chunkstr: The chunk string to which each rule should be 

1023 applied. 

1024 :type verbose: bool 

1025 :param verbose: Whether output should be verbose. 

1026 :rtype: None 

1027 """ 

1028 print("# Input:") 

1029 print(chunkstr) 

1030 for rule in self._rules: 

1031 rule.apply(chunkstr) 

1032 if verbose: 

1033 print("#", rule.descr() + " (" + repr(rule) + "):") 

1034 else: 

1035 print("#", rule.descr() + ":") 

1036 print(chunkstr) 

1037 

1038 def _notrace_apply(self, chunkstr): 

1039 """ 

1040 Apply each rule of this ``RegexpChunkParser`` to ``chunkstr``, in 

1041 turn. 

1042 

1043 :param chunkstr: The chunk string to which each rule should be 

1044 applied. 

1045 :type chunkstr: ChunkString 

1046 :rtype: None 

1047 """ 

1048 

1049 for rule in self._rules: 

1050 rule.apply(chunkstr) 

1051 

1052 def parse(self, chunk_struct, trace=None): 

1053 """ 

1054 :type chunk_struct: Tree 

1055 :param chunk_struct: the chunk structure to be (further) chunked 

1056 :type trace: int 

1057 :param trace: The level of tracing that should be used when 

1058 parsing a text. ``0`` will generate no tracing output; 

1059 ``1`` will generate normal tracing output; and ``2`` or 

1060 higher will generate verbose tracing output. This value 

1061 overrides the trace level value that was given to the 

1062 constructor. 

1063 :rtype: Tree 

1064 :return: a chunk structure that encodes the chunks in a given 

1065 tagged sentence. A chunk is a non-overlapping linguistic 

1066 group, such as a noun phrase. The set of chunks 

1067 identified in the chunk structure depends on the rules 

1068 used to define this ``RegexpChunkParser``. 

1069 """ 

1070 if len(chunk_struct) == 0: 

1071 print("Warning: parsing empty text") 

1072 return Tree(self._root_label, []) 

1073 

1074 try: 

1075 chunk_struct.label() 

1076 except AttributeError: 

1077 chunk_struct = Tree(self._root_label, chunk_struct) 

1078 

1079 # Use the default trace value? 

1080 if trace is None: 

1081 trace = self._trace 

1082 

1083 chunkstr = ChunkString(chunk_struct) 

1084 

1085 # Apply the sequence of rules to the chunkstring. 

1086 if trace: 

1087 verbose = trace > 1 

1088 self._trace_apply(chunkstr, verbose) 

1089 else: 

1090 self._notrace_apply(chunkstr) 

1091 

1092 # Use the chunkstring to create a chunk structure. 

1093 return chunkstr.to_chunkstruct(self._chunk_label) 

1094 

1095 def rules(self): 

1096 """ 

1097 :return: the sequence of rules used by ``RegexpChunkParser``. 

1098 :rtype: list(RegexpChunkRule) 

1099 """ 

1100 return self._rules 

1101 

1102 def __repr__(self): 

1103 """ 

1104 :return: a concise string representation of this 

1105 ``RegexpChunkParser``. 

1106 :rtype: str 

1107 """ 

1108 return "<RegexpChunkParser with %d rules>" % len(self._rules) 

1109 

1110 def __str__(self): 

1111 """ 

1112 :return: a verbose string representation of this ``RegexpChunkParser``. 

1113 :rtype: str 

1114 """ 

1115 s = "RegexpChunkParser with %d rules:\n" % len(self._rules) 

1116 margin = 0 

1117 for rule in self._rules: 

1118 margin = max(margin, len(rule.descr())) 

1119 if margin < 35: 

1120 format = " %" + repr(-(margin + 3)) + "s%s\n" 

1121 else: 

1122 format = " %s\n %s\n" 

1123 for rule in self._rules: 

1124 s += format % (rule.descr(), repr(rule)) 

1125 return s[:-1]

1133class RegexpParser(ChunkParserI): 

1134 r""" 

1135 A grammar based chunk parser. ``chunk.RegexpParser`` uses a set of 

1136 regular expression patterns to specify the behavior of the parser. 

1137 The chunking of the text is encoded using a ``ChunkString``, and 

1138 each rule acts by modifying the chunking in the ``ChunkString``. 

1139 The rules are all implemented using regular expression matching 

1140 and substitution. 

1141 

1142 A grammar contains one or more clauses in the following form:: 

1143 

1144 NP: 

1145 {<DT|JJ>} # chunk determiners and adjectives 

1146 }<[\.VI].*>+{ # strip any tag beginning with V, I, or . 

1147 <.*>}{<DT> # split a chunk at a determiner 

1148 <DT|JJ>{}<NN.*> # merge chunk ending with det/adj 

1149 # with one starting with a noun 

1150 

1151 The patterns of a clause are executed in order. An earlier 

1152 pattern may introduce a chunk boundary that prevents a later 

1153 pattern from executing. Sometimes an individual pattern will 

1154 match on multiple, overlapping extents of the input. As with 

1155 regular expression substitution more generally, the chunker will 

1156 identify the first match possible, then continue looking for matches 

1157 after this one has ended. 

1158 

1159 The clauses of a grammar are also executed in order. A cascaded 

1160 chunk parser is one having more than one clause. The maximum depth 

1161 of a parse tree created by this chunk parser is the same as the 

1162 number of clauses in the grammar. 

1163 

1164 When tracing is turned on, the comment portion of a line is displayed 

1165 each time the corresponding pattern is applied. 

1166 

1167 :type _start: str 

1168 :ivar _start: The start symbol of the grammar (the root node of 

1169 resulting trees) 

1170 :type _stages: int 

1171 :ivar _stages: The list of parsing stages corresponding to the grammar 

1172 

1173 """ 

1174 

1175 def __init__(self, grammar, root_label="S", loop=1, trace=0): 

1176 """ 

1177 Create a new chunk parser, from the given start state 

1178 and set of chunk patterns. 

1179 

1180 :param grammar: The grammar, or a list of RegexpChunkParser objects 

1181 :type grammar: str or list(RegexpChunkParser) 

1182 :param root_label: The top node of the tree being created 

1183 :type root_label: str or Nonterminal 

1184 :param loop: The number of times to run through the patterns 

1185 :type loop: int 

1186 :type trace: int 

1187 :param trace: The level of tracing that should be used when 

1188 parsing a text. ``0`` will generate no tracing output; 

1189 ``1`` will generate normal tracing output; and ``2`` or 

1190 higher will generate verbose tracing output. 

1191 """ 

1192 self._trace = trace 

1193 self._stages = [] 

1194 self._grammar = grammar 

1195 self._loop = loop 

1196 

1197 if isinstance(grammar, str): 

1198 self._read_grammar(grammar, root_label, trace) 

1199 else: 

1200 # Make sur the grammar looks like it has the right type: 

1201 type_err = ( 

1202 "Expected string or list of RegexpChunkParsers " "for the grammar." 

1203 ) 

1204 try: 

1205 grammar = list(grammar) 

1206 except BaseException as e: 

1207 raise TypeError(type_err) from e 

1208 for elt in grammar: 

1209 if not isinstance(elt, RegexpChunkParser): 

1210 raise TypeError(type_err) 

1211 self._stages = grammar 

1212 

1213 def _read_grammar(self, grammar, root_label, trace): 

1214 """ 

1215 Helper function for __init__: read the grammar if it is a 

1216 string. 

1217 """ 

1218 rules = [] 

1219 lhs = None 

1220 for line in grammar.split("\n"): 

1221 line = line.strip() 

1222 

1223 # New stage begins if there's an unescaped ':' 

1224 m = re.match("(?P<nonterminal>(\\.|[^:])*)(:(?P<rule>.*))", line) 

1225 if m: 

1226 # Record the stage that we just completed. 

1227 self._add_stage(rules, lhs, root_label, trace) 

1228 # Start a new stage. 

1229 lhs = m.group("nonterminal").strip() 

1230 rules = [] 

1231 line = m.group("rule").strip() 

1232 

1233 # Skip blank & comment-only lines 

1234 if line == "" or line.startswith("#"): 

1235 continue 

1236 

1237 # Add the rule 

1238 rules.append(RegexpChunkRule.fromstring(line)) 

1239 

1240 # Record the final stage 

1241 self._add_stage(rules, lhs, root_label, trace) 

1242 

1243 def _add_stage(self, rules, lhs, root_label, trace): 

1244 """ 

1245 Helper function for __init__: add a new stage to the parser. 

1246 """ 

1247 if rules != []: 

1248 if not lhs: 

1249 raise ValueError("Expected stage marker (eg NP:)") 

1250 parser = RegexpChunkParser( 

1251 rules, chunk_label=lhs, root_label=root_label, trace=trace 

1252 ) 

1253 self._stages.append(parser) 

1254 

1255 def parse(self, chunk_struct, trace=None): 

1256 """ 

1257 Apply the chunk parser to this input. 

1258 

1259 :type chunk_struct: Tree 

1260 :param chunk_struct: the chunk structure to be (further) chunked 

1261 (this tree is modified, and is also returned) 

1262 :type trace: int 

1263 :param trace: The level of tracing that should be used when 

1264 parsing a text. ``0`` will generate no tracing output; 

1265 ``1`` will generate normal tracing output; and ``2`` or 

1266 higher will generate verbose tracing output. This value 

1267 overrides the trace level value that was given to the 

1268 constructor. 

1269 :return: the chunked output. 

1270 :rtype: Tree 

1271 """ 

1272 if trace is None: 

1273 trace = self._trace 

1274 for i in range(self._loop): 

1275 for parser in self._stages: 

1276 chunk_struct = parser.parse(chunk_struct, trace=trace) 

1277 return chunk_struct 

1278 

1279 def __repr__(self): 

1280 """ 

1281 :return: a concise string representation of this ``chunk.RegexpParser``. 

1282 :rtype: str 

1283 """ 

1284 return "<chunk.RegexpParser with %d stages>" % len(self._stages) 

1285 

1286 def __str__(self): 

1287 """ 

1288 :return: a verbose string representation of this 

1289 ``RegexpParser``. 

1290 :rtype: str 

1291 """ 

1292 s = "chunk.RegexpParser with %d stages:\n" % len(self._stages) 

1293 margin = 0 

1294 for parser in self._stages: 

1295 s += "%s\n" % parser 

1296 return s[:-1]

1304def demo_eval(chunkparser, text): 

1305 """ 

1306 Demonstration code for evaluating a chunk parser, using a 

1307 ``ChunkScore``. This function assumes that ``text`` contains one 

1308 sentence per line, and that each sentence has the form expected by 

1309 ``tree.chunk``. It runs the given chunk parser on each sentence in 

1310 the text, and scores the result. It prints the final score 

1311 (precision, recall, and f-measure); and reports the set of chunks 

1312 that were missed and the set of chunks that were incorrect. (At 

1313 most 10 missing chunks and 10 incorrect chunks are reported). 

1314 

1315 :param chunkparser: The chunkparser to be tested 

1316 :type chunkparser: ChunkParserI 

1317 :param text: The chunked tagged text that should be used for 

1318 evaluation. 

1319 :type text: str 

1320 """ 

1321 from nltk import chunk 

1322 from nltk.tree import Tree 

1323 

1324 # Evaluate our chunk parser. 

1325 chunkscore = chunk.ChunkScore() 

1326 

1327 for sentence in text.split("\n"): 

1328 print(sentence) 

1329 sentence = sentence.strip() 

1330 if not sentence: 

1331 continue 

1332 gold = chunk.tagstr2tree(sentence) 

1333 tokens = gold.leaves() 

1334 test = chunkparser.parse(Tree("S", tokens), trace=1) 

1335 chunkscore.score(gold, test) 

1336 print() 

1337 

1338 print("/" + ("=" * 75) + "\\") 

1339 print("Scoring", chunkparser) 

1340 print("-" * 77) 

1341 print("Precision: %5.1f%%" % (chunkscore.precision() * 100), " " * 4, end=" ") 

1342 print("Recall: %5.1f%%" % (chunkscore.recall() * 100), " " * 6, end=" ") 

1343 print("F-Measure: %5.1f%%" % (chunkscore.f_measure() * 100)) 

1344 

1345 # Missed chunks. 

1346 if chunkscore.missed(): 

1347 print("Missed:") 

1348 missed = chunkscore.missed() 

1349 for chunk in missed[:10]: 

1350 print(" ", " ".join(map(str, chunk))) 

1351 if len(chunkscore.missed()) > 10: 

1352 print(" ...") 

1353 

1354 # Incorrect chunks. 

1355 if chunkscore.incorrect(): 

1356 print("Incorrect:") 

1357 incorrect = chunkscore.incorrect() 

1358 for chunk in incorrect[:10]: 

1359 print(" ", " ".join(map(str, chunk))) 

1360 if len(chunkscore.incorrect()) > 10: 

1361 print(" ...") 

1362 

1363 print("\\" + ("=" * 75) + "/") 

1364 print()
