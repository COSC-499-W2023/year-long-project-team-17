16def sentence_ribes(references, hypothesis, alpha=0.25, beta=0.10): 

17 """ 

18 The RIBES (Rank-based Intuitive Bilingual Evaluation Score) from 

19 Hideki Isozaki, Tsutomu Hirao, Kevin Duh, Katsuhito Sudoh and 

20 Hajime Tsukada. 2010. "Automatic Evaluation of Translation Quality for 

21 Distant Language Pairs". In Proceedings of EMNLP. 

22 https://www.aclweb.org/anthology/D/D10/D10-1092.pdf 

23 

24 The generic RIBES scores used in shared task, e.g. Workshop for 

25 Asian Translation (WAT) uses the following RIBES calculations: 

26 

27 RIBES = kendall_tau * (alpha**p1) * (beta**bp) 

28 

29 Please note that this re-implementation differs from the official 

30 RIBES implementation and though it emulates the results as describe 

31 in the original paper, there are further optimization implemented 

32 in the official RIBES script. 

33 

34 Users are encouraged to use the official RIBES script instead of this 

35 implementation when evaluating your machine translation system. Refer 

36 to https://www.kecl.ntt.co.jp/icl/lirg/ribes/ for the official script. 

37 

38 :param references: a list of reference sentences 

39 :type references: list(list(str)) 

40 :param hypothesis: a hypothesis sentence 

41 :type hypothesis: list(str) 

42 :param alpha: hyperparameter used as a prior for the unigram precision. 

43 :type alpha: float 

44 :param beta: hyperparameter used as a prior for the brevity penalty. 

45 :type beta: float 

46 :return: The best ribes score from one of the references. 

47 :rtype: float 

48 """ 

49 best_ribes = -1.0 

50 # Calculates RIBES for each reference and returns the best score. 

51 for reference in references: 

52 # Collects the *worder* from the ranked correlation alignments. 

53 worder = word_rank_alignment(reference, hypothesis) 

54 nkt = kendall_tau(worder) 

55 

56 # Calculates the brevity penalty 

57 bp = min(1.0, math.exp(1.0 - len(reference) / len(hypothesis))) 

58 

59 # Calculates the unigram precision, *p1* 

60 p1 = len(worder) / len(hypothesis) 

61 

62 _ribes = nkt * (p1 ** alpha) * (bp ** beta) 

63 

64 if _ribes > best_ribes: # Keeps the best score. 

65 best_ribes = _ribes 

66 

67 return best_ribes

70def corpus_ribes(list_of_references, hypotheses, alpha=0.25, beta=0.10): 

71 """ 

72 This function "calculates RIBES for a system output (hypothesis) with 

73 multiple references, and returns "best" score among multi-references and 

74 individual scores. The scores are corpus-wise, i.e., averaged by the number 

75 of sentences." (c.f. RIBES version 1.03.1 code). 

76 

77 Different from BLEU's micro-average precision, RIBES calculates the 

78 macro-average precision by averaging the best RIBES score for each pair of 

79 hypothesis and its corresponding references 

80 

81 >>> hyp1 = ['It', 'is', 'a', 'guide', 'to', 'action', 'which', 

82 ... 'ensures', 'that', 'the', 'military', 'always', 

83 ... 'obeys', 'the', 'commands', 'of', 'the', 'party'] 

84 >>> ref1a = ['It', 'is', 'a', 'guide', 'to', 'action', 'that', 

85 ... 'ensures', 'that', 'the', 'military', 'will', 'forever', 

86 ... 'heed', 'Party', 'commands'] 

87 >>> ref1b = ['It', 'is', 'the', 'guiding', 'principle', 'which', 

88 ... 'guarantees', 'the', 'military', 'forces', 'always', 

89 ... 'being', 'under', 'the', 'command', 'of', 'the', 'Party'] 

90 >>> ref1c = ['It', 'is', 'the', 'practical', 'guide', 'for', 'the', 

91 ... 'army', 'always', 'to', 'heed', 'the', 'directions', 

92 ... 'of', 'the', 'party'] 

93 

94 >>> hyp2 = ['he', 'read', 'the', 'book', 'because', 'he', 'was', 

95 ... 'interested', 'in', 'world', 'history'] 

96 >>> ref2a = ['he', 'was', 'interested', 'in', 'world', 'history', 

97 ... 'because', 'he', 'read', 'the', 'book'] 

98 

99 >>> list_of_references = [[ref1a, ref1b, ref1c], [ref2a]] 

100 >>> hypotheses = [hyp1, hyp2] 

101 >>> round(corpus_ribes(list_of_references, hypotheses),4) 

102 0.3597 

103 

104 :param references: a corpus of lists of reference sentences, w.r.t. hypotheses 

105 :type references: list(list(list(str))) 

106 :param hypotheses: a list of hypothesis sentences 

107 :type hypotheses: list(list(str)) 

108 :param alpha: hyperparameter used as a prior for the unigram precision. 

109 :type alpha: float 

110 :param beta: hyperparameter used as a prior for the brevity penalty. 

111 :type beta: float 

112 :return: The best ribes score from one of the references. 

113 :rtype: float 

114 """ 

115 corpus_best_ribes = 0.0 

116 # Iterate through each hypothesis and their corresponding references. 

117 for references, hypothesis in zip(list_of_references, hypotheses): 

118 corpus_best_ribes += sentence_ribes(references, hypothesis, alpha, beta) 

119 return corpus_best_ribes / len(hypotheses)

122def position_of_ngram(ngram, sentence): 

123 """ 

124 This function returns the position of the first instance of the ngram 

125 appearing in a sentence. 

126 

127 Note that one could also use string as follows but the code is a little 

128 convoluted with type casting back and forth: 

129 

130 char_pos = ' '.join(sent)[:' '.join(sent).index(' '.join(ngram))] 

131 word_pos = char_pos.count(' ') 

132 

133 Another way to conceive this is: 

134 

135 return next(i for i, ng in enumerate(ngrams(sentence, len(ngram))) 

136 if ng == ngram) 

137 

138 :param ngram: The ngram that needs to be searched 

139 :type ngram: tuple 

140 :param sentence: The list of tokens to search from. 

141 :type sentence: list(str) 

142 """ 

143 # Iterates through the ngrams in sentence. 

144 for i, sublist in enumerate(ngrams(sentence, len(ngram))): 

145 # Returns the index of the word when ngram matches. 

146 if ngram == sublist: 

147 return i

150def word_rank_alignment(reference, hypothesis, character_based=False): 

151 """ 

152 This is the word rank alignment algorithm described in the paper to produce 

153 the *worder* list, i.e. a list of word indices of the hypothesis word orders 

154 w.r.t. the list of reference words. 

155 

156 Below is (H0, R0) example from the Isozaki et al. 2010 paper, 

157 note the examples are indexed from 1 but the results here are indexed from 0: 

158 

159 >>> ref = str('he was interested in world history because he ' 

160 ... 'read the book').split() 

161 >>> hyp = str('he read the book because he was interested in world ' 

162 ... 'history').split() 

163 >>> word_rank_alignment(ref, hyp) 

164 [7, 8, 9, 10, 6, 0, 1, 2, 3, 4, 5] 

165 

166 The (H1, R1) example from the paper, note the 0th index: 

167 

168 >>> ref = 'John hit Bob yesterday'.split() 

169 >>> hyp = 'Bob hit John yesterday'.split() 

170 >>> word_rank_alignment(ref, hyp) 

171 [2, 1, 0, 3] 

172 

173 Here is the (H2, R2) example from the paper, note the 0th index here too: 

174 

175 >>> ref = 'the boy read the book'.split() 

176 >>> hyp = 'the book was read by the boy'.split() 

177 >>> word_rank_alignment(ref, hyp) 

178 [3, 4, 2, 0, 1] 

179 

180 :param reference: a reference sentence 

181 :type reference: list(str) 

182 :param hypothesis: a hypothesis sentence 

183 :type hypothesis: list(str) 

184 """ 

185 worder = [] 

186 hyp_len = len(hypothesis) 

187 # Stores a list of possible ngrams from the reference sentence. 

188 # This is used for matching context window later in the algorithm. 

189 ref_ngrams = [] 

190 hyp_ngrams = [] 

191 for n in range(1, len(reference) + 1): 

192 for ng in ngrams(reference, n): 

193 ref_ngrams.append(ng) 

194 for ng in ngrams(hypothesis, n): 

195 hyp_ngrams.append(ng) 

196 for i, h_word in enumerate(hypothesis): 

197 # If word is not in the reference, continue. 

198 if h_word not in reference: 

199 continue 

200 # If we can determine one-to-one word correspondence for unigrams that 

201 # only appear once in both the reference and hypothesis. 

202 elif hypothesis.count(h_word) == reference.count(h_word) == 1: 

203 worder.append(reference.index(h_word)) 

204 else: 

205 max_window_size = max(i, hyp_len - i + 1) 

206 for window in range(1, max_window_size): 

207 if i + window < hyp_len: # If searching the right context is possible. 

208 # Retrieve the right context window. 

209 right_context_ngram = tuple(islice(hypothesis, i, i + window + 1)) 

210 num_times_in_ref = ref_ngrams.count(right_context_ngram) 

211 num_times_in_hyp = hyp_ngrams.count(right_context_ngram) 

212 # If ngram appears only once in both ref and hyp. 

213 if num_times_in_ref == num_times_in_hyp == 1: 

214 # Find the position of ngram that matched the reference. 

215 pos = position_of_ngram(right_context_ngram, reference) 

216 worder.append(pos) # Add the positions of the ngram. 

217 break 

218 if window <= i: # If searching the left context is possible. 

219 # Retrieve the left context window. 

220 left_context_ngram = tuple(islice(hypothesis, i - window, i + 1)) 

221 num_times_in_ref = ref_ngrams.count(left_context_ngram) 

222 num_times_in_hyp = hyp_ngrams.count(left_context_ngram) 

223 if num_times_in_ref == num_times_in_hyp == 1: 

224 # Find the position of ngram that matched the reference. 

225 pos = position_of_ngram(left_context_ngram, reference) 

226 # Add the positions of the ngram. 

227 worder.append(pos + len(left_context_ngram) - 1) 

228 break 

229 return worder

232def find_increasing_sequences(worder): 

233 """ 

234 Given the *worder* list, this function groups monotonic +1 sequences. 

235 

236 >>> worder = [7, 8, 9, 10, 6, 0, 1, 2, 3, 4, 5] 

237 >>> list(find_increasing_sequences(worder)) 

238 [(7, 8, 9, 10), (0, 1, 2, 3, 4, 5)] 

239 

240 :param worder: The worder list output from word_rank_alignment 

241 :param type: list(int) 

242 """ 

243 items = iter(worder) 

244 a, b = None, next(items, None) 

245 result = [b] 

246 while b is not None: 

247 a, b = b, next(items, None) 

248 if b is not None and a + 1 == b: 

249 result.append(b) 

250 else: 

251 if len(result) > 1: 

252 yield tuple(result) 

253 result = [b]

256def kendall_tau(worder, normalize=True): 

257 """ 

258 Calculates the Kendall's Tau correlation coefficient given the *worder* 

259 list of word alignments from word_rank_alignment(), using the formula: 

260 

261 tau = 2 * num_increasing_pairs / num_possible_pairs -1 

262 

263 Note that the no. of increasing pairs can be discontinuous in the *worder* 

264 list and each each increasing sequence can be tabulated as choose(len(seq), 2) 

265 no. of increasing pairs, e.g. 

266 

267 >>> worder = [7, 8, 9, 10, 6, 0, 1, 2, 3, 4, 5] 

268 >>> number_possible_pairs = choose(len(worder), 2) 

269 >>> round(kendall_tau(worder, normalize=False),3) 

270 -0.236 

271 >>> round(kendall_tau(worder),3) 

272 0.382 

273 

274 :param worder: The worder list output from word_rank_alignment 

275 :type worder: list(int) 

276 :param normalize: Flag to indicate normalization to between 0.0 and 1.0. 

277 :type normalize: boolean 

278 :return: The Kendall's Tau correlation coefficient. 

279 :rtype: float 

280 """ 

281 worder_len = len(worder) 

282 # With worder_len < 2, `choose(worder_len, 2)` will be 0. 

283 # As we divide by this, it will give a ZeroDivisionError. 

284 # To avoid this, we can just return the lowest possible score. 

285 if worder_len < 2: 

286 tau = -1 

287 else: 

288 # Extract the groups of increasing/monotonic sequences. 

289 increasing_sequences = find_increasing_sequences(worder) 

290 # Calculate no. of increasing_pairs in *worder* list. 

291 num_increasing_pairs = sum(choose(len(seq), 2) for seq in increasing_sequences) 

292 # Calculate no. of possible pairs. 

293 num_possible_pairs = choose(worder_len, 2) 

294 # Kendall's Tau computation. 

295 tau = 2 * num_increasing_pairs / num_possible_pairs - 1 

296 if normalize: # If normalized, the tau output falls between 0.0 to 1.0 

297 return (tau + 1) / 2 

298 else: # Otherwise, the tau outputs falls between -1.0 to +1.0 

299 return tau

302def spearman_rho(worder, normalize=True): 

303 """ 

304 Calculates the Spearman's Rho correlation coefficient given the *worder* 

305 list of word alignment from word_rank_alignment(), using the formula: 

306 

307 rho = 1 - sum(d**2) / choose(len(worder)+1, 3) 

308 

309 Given that d is the sum of difference between the *worder* list of indices 

310 and the original word indices from the reference sentence. 

311 

312 Using the (H0,R0) and (H5, R5) example from the paper 

313 

314 >>> worder = [7, 8, 9, 10, 6, 0, 1, 2, 3, 4, 5] 

315 >>> round(spearman_rho(worder, normalize=False), 3) 

316 -0.591 

317 >>> round(spearman_rho(worder), 3) 

318 0.205 

319 

320 :param worder: The worder list output from word_rank_alignment 

321 :param type: list(int) 

322 """ 

323 worder_len = len(worder) 

324 sum_d_square = sum((wi - i) ** 2 for wi, i in zip(worder, range(worder_len))) 

325 rho = 1 - sum_d_square / choose(worder_len + 1, 3) 

326 

327 if normalize: # If normalized, the rho output falls between 0.0 to 1.0 

328 return (rho + 1) / 2 

329 else: # Otherwise, the rho outputs falls between -1.0 to +1.0 

330 return rho
