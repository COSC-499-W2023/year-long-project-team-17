26class CT_Body(BaseOxmlElement): 

27 """ 

28 ``<w:body>``, the container element for the main document story in 

29 ``document.xml``. 

30 """ 

31 p = ZeroOrMore('w:p', successors=('w:sectPr',))

33 sectPr = ZeroOrOne('w:sectPr', successors=())

35 def add_section_break(self): 

36 """Return `w:sectPr` element for new section added at end of document. 

37 

38 The last `w:sectPr` becomes the second-to-last, with the new `w:sectPr` being an 

39 exact clone of the previous one, except that all header and footer references 

40 are removed (and are therefore now "inherited" from the prior section). 

41 

42 A copy of the previously-last `w:sectPr` will now appear in a new `w:p` at the 

43 end of the document. The returned `w:sectPr` is the sentinel `w:sectPr` for the 

44 document (and as implemented, *is* the prior sentinel `w:sectPr` with headers 

45 and footers removed). 

46 """ 

47 # ---get the sectPr at file-end, which controls last section (sections[-1])--- 

48 sentinel_sectPr = self.get_or_add_sectPr() 

49 # ---add exact copy to new `w:p` element; that is now second-to last section--- 

50 self.add_p().set_sectPr(sentinel_sectPr.clone()) 

51 # ---remove any header or footer references from "new" last section--- 

52 for hdrftr_ref in sentinel_sectPr.xpath("w:headerReference|w:footerReference"): 

53 sentinel_sectPr.remove(hdrftr_ref) 

54 # ---the sentinel `w:sectPr` now controls the new last section--- 

55 return sentinel_sectPr

58 """ 

59 Remove all content child elements from this <w:body> element. Leave 

60 the <w:sectPr> element if it is present. 

61 """ 

62 if self.sectPr is not None: 

63 content_elms = self[:-1] 

64 else: 

65 content_elms = self[:] 

66 for content_elm in content_elms: 

67 self.remove(content_elm)
